7   import java.io.IOException;
8   import java.security.SecureRandom;
9   import java.util.Arrays;
10  import javax.annotation.concurrent.NotThreadSafe;
11  import net.java.truecommons.io.DecoratingOutputStream;
12  import net.java.truecommons.io.LittleEndianOutputStream;
13  import net.java.truecommons.key.spec.KeyStrength;
14  import net.java.truevfs.comp.zip.crypto.BufferedPartialBlockCipher;
15  import net.java.truevfs.comp.zip.crypto.CipherOutputStream;
16  import org.bouncycastle.crypto.BufferedBlockCipher;
17  import org.bouncycastle.crypto.Mac;
18  import org.bouncycastle.crypto.PBEParametersGenerator;
19  import org.bouncycastle.crypto.digests.SHA1Digest;
20  import org.bouncycastle.crypto.generators.PKCS5S2ParametersGenerator;
21  import org.bouncycastle.crypto.io.MacOutputStream;
22  import org.bouncycastle.crypto.macs.HMac;
23  import org.bouncycastle.crypto.params.KeyParameter;
24  import org.bouncycastle.crypto.params.ParametersWithIV;
25  import org.bouncycastle.util.io.TeeOutputStream;

27  /**
28   * Encrypts ZIP entry contents according the WinZip AES specification.
29   *
30   * @see     <a href="http://www.winzip.com/win/en/aes_info.htm">AES Encryption Information: Encryption Specification AE-1 and AE-2 (WinZip Computing, S.L.)</a>
31   * @see     <a href="http://www.winzip.com/win/en/aes_tips.htm">AES Coding Tips for Developers (WinZip Computing, S.L.)</a>
32   * @see     <a href="http://www.gladman.me.uk/cryptography_technology/fileencrypt/">A Password Based File Encyption Utility (Dr. Gladman)</a>
33   * @see     <a href="http://www.ietf.org/rfc/rfc2898.txt">RFC 2898: PKCS #5: Password-Based Cryptography Specification Version 2.0 (IETF et al.)</a>
34   * @see     AbstractZipOutputStream$WinZipAesOutputMethod
35   * @author  Christian Schlichtherle
36   */
37  @NotThreadSafe
38  final class WinZipAesOutputStream extends DecoratingOutputStream {
39  
40      /**
41       * The iteration count for the derived keys of the cipher, KLAC and MAC.
42       */
43      static final int ITERATION_COUNT = 1000;
44  
45      /**
46       * The block size of the Advanced Encryption Specification (AES) Algorithm
47       * in bits ({@value #AES_BLOCK_SIZE_BITS}).
48       */
49      static final int AES_BLOCK_SIZE_BITS = 128;
50  
51      static final int PWD_VERIFIER_BITS = 16;
52  
53      /** The message authentication code (MAC) output stream. */
54      private final MacOutputStream mos;
55  
56      /**
57       * The low level data output stream.
58       * Used for writing the header and footer.
59       **/
60      private final LittleEndianOutputStream leos;
61  
62      WinZipAesOutputStream(
63              final WinZipAesEntryParameters param,
64              final LittleEndianOutputStream leos)
65      throws IOException {
66          assert null != param;
67          try {
68              assert null != leos;
69  
70              // Init key strength.
71              final KeyStrength keyStrength = param.getKeyStrength();
72              final int keyStrengthBits = keyStrength.getBits();
73              final int keyStrengthBytes = keyStrength.getBytes();
74  
75              // Shake the salt.
76              final byte[] salt = new byte[keyStrengthBytes / 2];
77              new SecureRandom().nextBytes(salt);
78  
79              // Init password.
80              final byte[] passwd = param.getWritePassword();
81  
82              // Derive cipher and MAC parameters.
83              final PBEParametersGenerator gen = new PKCS5S2ParametersGenerator();
84              gen.init(passwd, salt, ITERATION_COUNT);
85              // Here comes the strange part about WinZip AES encryption:
86              // Its unorthodox use of the Password-Based Key Derivation
87              // Function 2 (PBKDF2) of PKCS #5 V2.0 alias RFC 2898.
88              // Yes, the password verifier is only a 16 bit value.
89              // So we must use the MAC for password verification, too.
90              assert AES_BLOCK_SIZE_BITS <= keyStrengthBits;
91              final KeyParameter keyParam =
92                      (KeyParameter) gen.generateDerivedParameters(
93                          2 * keyStrengthBits + PWD_VERIFIER_BITS);
94              Arrays.fill(passwd, (byte) 0); // must not wipe before generator use!
95  
96              // Can you believe they "forgot" the nonce in the CTR mode IV?! :-(
97              final byte[] ctrIv = new byte[AES_BLOCK_SIZE_BITS / 8];
98              final ParametersWithIV aesCtrParam = new ParametersWithIV(
99                      new KeyParameter(keyParam.getKey(), 0, keyStrengthBytes),
100                     ctrIv); // yes, the IV is an array of zero bytes!
101             final KeyParameter sha1HMacParam = new KeyParameter(
102                     keyParam.getKey(),
103                     keyStrengthBytes,
104                     keyStrengthBytes);
105 
106             // Init cipher and stream.
107             final BufferedBlockCipher
108                     cipher = new BufferedPartialBlockCipher(new WinZipAesCipher());
109             cipher.init(true, aesCtrParam);
110 
111             // Init MAC.
112             final Mac mac = new HMac(new SHA1Digest());
113             mac.init(sha1HMacParam);
114 
115             // Init chain of output streams as Encrypt-then-MAC.
116             this.leos = leos;
117             mos = new MacOutputStream(mac);
118             out = new CipherOutputStream(cipher,
119                     new TeeOutputStream(leos, mos));
120 
121             // Write header.
122             leos.write(salt);
123             writePasswordVerifier(param, keyParam);
124         } catch (final Throwable ex) {
125             try {
126                 leos.close();
127             } catch (final Throwable ex2) {
128                 ex.addSuppressed(ex2);
129             }
130             throw ex;
131         }
132     }
133 
134     private void writePasswordVerifier(WinZipAesEntryParameters param, KeyParameter keyParam)
135     throws IOException {
136         leos.write(
137                 keyParam.getKey(),
138                 2 * param.getKeyStrength().getBytes(),
139                 PWD_VERIFIER_BITS / 8);
140     }
141 
142     void finish() throws IOException {
143         // Flush partial block to out, if any.
144         ((CipherOutputStream) out).finish();
145 
146         // Compute and write the first half of the MAC to the footer.
147         final byte[] buf = mos.getMac();
148         leos.write(buf, 0, buf.length / 2);
149     }
150 }
