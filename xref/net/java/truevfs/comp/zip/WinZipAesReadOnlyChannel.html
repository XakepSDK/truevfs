32  /**
33   * Decrypts ZIP entry contents according the WinZip AES specification.
34   * <p>
35   * Note that this channel implements its own virtual position.
36   *
37   * @see     <a href="http://www.winzip.com/win/en/aes_info.htm">AES Encryption Information: Encryption Specification AE-1 and AE-2 (WinZip Computing, S.L.)</a>
38   * @see     <a href="http://www.winzip.com/win/en/aes_tips.htm">AES Coding Tips for Developers (WinZip Computing, S.L.)</a>
39   * @see     <a href="http://www.gladman.me.uk/cryptography_technology/fileencrypt/">A Password Based File Encyption Utility (Dr. Gladman)</a>
40   * @see     <a href="http://www.ietf.org/rfc/rfc2898.txt">RFC 2898: PKCS #5: Password-Based Cryptography Specification Version 2.0 (IETF et al.)</a>
41   * @see     RawZipOutputStream$WinZipAesOutputMethod
42   * @author  Christian Schlichtherle
43   */
44  @NotThreadSafe
45  final class WinZipAesReadOnlyChannel extends ReadOnlyChannel {
46  
47      private static final int MAC_SIZE = newMac().getMacSize();
48  
49      private static Mac newMac() { return new HMac(new SHA1Digest()); }
50  
51      private final ByteBuffer authenticationCode;
52  
53      /**
54       * The key parameter required to init the SHA-1 Message Authentication
55       * Code (HMAC).
56       */
57      private final KeyParameter sha1MacParam;
58  
59      private final ZipEntry entry;
60  
61      @CreatesObligation
62      WinZipAesReadOnlyChannel(
63              final @WillCloseWhenClosed SeekableByteChannel channel,
64              final WinZipAesEntryParameters param)
65      throws IOException {
66          super(channel);
67  
68          // Init WinZip AES extra field.
69          final ZipEntry entry = param.getEntry();
70          assert entry.isEncrypted();
71          final WinZipAesExtraField
72                  field = (WinZipAesExtraField) entry.getExtraField(WINZIP_AES_ID);
73          if (null == field)
74              throw new ZipCryptoException(entry.getName() + " (missing extra field for WinZip AES entry)");
75  
76          // Get key strength.
77          final AesKeyStrength keyStrength = field.getKeyStrength();
78          final int keyStrengthBits = keyStrength.getBits();
79          final int keyStrengthBytes = keyStrength.getBytes();
80  
81          // Load salt.
82          final ByteBuffer salt = MutableBuffer
83                  .allocate(keyStrengthBytes / 2)
84                  .load(channel.position(0))
85                  .buffer();
86  
87          // Load password verification value.
88          final ByteBuffer passwdVerifier = MutableBuffer
89                  .allocate(PWD_VERIFIER_BITS / 8)
90                  .load(channel)
91                  .buffer();
92  
93          // Init MAC and authentication code.
94          final MutableBuffer footer = MutableBuffer.allocate(MAC_SIZE / 2);
95  
96          // Init start, end and size of encrypted data.
97          final long start = channel.position();
98          final long end = channel.size() - footer.remaining();
99          final long size = end - start;
100         if (0 > size) {
101             // Wrap an EOFException so that RawReadOnlyChannel can identify this issue.
102             throw new ZipCryptoException(entry.getName()
103                     + " (false positive WinZip AES entry is too short)",
104                     new EOFException());
105         }
106 
107         // Load authentication code.
108         footer.load(channel.position(end));
109         if (channel.position() != channel.size()) {
110             // This should never happen unless someone is writing to the
111             // end of the file concurrently!
112             throw new ZipCryptoException(
113                     "Expected end of file after WinZip AES authentication code!");
114         }
115         authenticationCode = footer.buffer();
116 
117         // Derive cipher and MAC parameters.
118         final PBEParametersGenerator gen = new PKCS5S2ParametersGenerator();
119         KeyParameter keyParam;
120         ParametersWithIV aesCtrParam;
121         KeyParameter sha1MacParam;
122         long lastTry = 0; // don't enforce suspension on first prompt!
123         do {
124             final byte[] passwd = param.getReadPassword(0 != lastTry);
125             assert null != passwd;
126 
127             gen.init(passwd, salt.array(), ITERATION_COUNT);
128             // Here comes the strange part about WinZip AES encryption:
129             // Its unorthodox use of the Password-Based Key Derivation
130             // Function 2 (PBKDF2) of PKCS #5 V2.0 alias RFC 2898.
131             // Yes, the password verifier is only a 16 bit value.
132             // So we must use the MAC for password verification, too.
133             assert AES_BLOCK_SIZE_BITS <= keyStrengthBits;
134             keyParam = (KeyParameter) gen.generateDerivedParameters(
135                     2 * keyStrengthBits + PWD_VERIFIER_BITS);
136             Arrays.fill(passwd, (byte) 0);
137 
138             // Can you believe they "forgot" the nonce in the CTR mode IV?! :-(
139             final byte[] ctrIv = new byte[AES_BLOCK_SIZE_BITS / 8];
140             aesCtrParam = new ParametersWithIV(
141                     new KeyParameter(keyParam.getKey(), 0, keyStrengthBytes),
142                     ctrIv); // yes, the IV is an array of zero bytes!
143             sha1MacParam = new KeyParameter(
144                     keyParam.getKey(),
145                     keyStrengthBytes,
146                     keyStrengthBytes);
147 
148             lastTry = SuspensionPenalty.enforce(lastTry);
149 
150             // Verify password.
151         } while (!passwdVerifier.equals(ByteBuffer.wrap(
152                 keyParam.getKey()).position(2 * keyStrengthBytes)));
153 
154         // Init parameters and entry for authenticate().
155         this.sha1MacParam = sha1MacParam;
156         this.entry = entry;
157 
158         // Init cipher and channel.
159         final SeekableBlockCipher cipher = new WinZipAesCipher();
160         cipher.init(false, aesCtrParam);
161         this.channel = new CipherReadOnlyChannel(cipher,
162                 new IntervalReadOnlyChannel(channel.position(start), size));
163 
164         // Commit key strength.
165         param.setKeyStrength(keyStrength);
166     }
167 
168     /**
169      * Authenticates all encrypted data in this read-only channel.
170      * This method can get called multiple times to detect if the file has been
171      * tampered with meanwhile.
172      *
173      * @throws ZipAuthenticationException If the computed MAC does not match
174      *         the MAC declared in the WinZip AES entry.
175      * @throws IOException On any I/O related issue.
176      */
177     void authenticate() throws IOException {
178         final Mac mac = newMac();
179         mac.init(sha1MacParam);
180         final byte[] buf = ((CipherReadOnlyChannel) channel).mac(mac);
181         if (!authenticationCode.equals(ByteBuffer.wrap(buf, 0, buf.length / 2)))
182             throw new ZipAuthenticationException(entry.getName()
183                     + " (authenticated WinZip AES entry content has been tampered with)");
184     }
185 }
