5   package net.java.truevfs.comp.zip;
6   
7   import net.java.truevfs.comp.zip.crypto.CtrBlockCipher;
8   import org.bouncycastle.crypto.engines.AESEngine;
9   
10  /**
11   * Implements Counter (CTR) mode (alias Segmented Integer Counter - SIC)
12   * on top of an AES engine.
13   * This class is almost identical to {@link CtrBlockCipher} except that
14   * the block counter is incremented <em>before</em> updating the cipher input
15   * in <em>little endian</em> order.
16   *
17   * @see    <a href="http://www.gladman.me.uk/cryptography_technology/fileencrypt/">A Password Based File Encyption Utility (Dr. Gladman)</a>
18   * @author Christian Schlichtherle
19   */
20  final class WinZipAesCipher extends CtrBlockCipher {
21  
22      /**
23       * Constructs a new block cipher mode for use with WinZip AES.
24       * This constructor uses an {@link AESEngine} as the underlying block
25       * cipher.
26       */
27      WinZipAesCipher() {
28          super(new AESEngine());
29      }
30  
31      @Override
32      protected void incCounter() {
33          final int blockSize = this.blockSize;
34          long blockCounter = ++this.blockCounter; // pre-increment the block counter!
35          for (int i = 0; i < blockSize; i++) { // little endian order!
36              blockCounter += IV[i] & 0xff;
37              cipherIn[i] = (byte) blockCounter;
38              blockCounter >>>= 8;
39          }
40      }
41  }
