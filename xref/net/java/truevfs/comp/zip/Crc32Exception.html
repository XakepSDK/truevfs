5   package net.java.truevfs.comp.zip;
6   
7   import java.util.zip.ZipException;
8   
9   /**
10   * Thrown to indicate a CRC32 mismatch between the declared value in the
11   * Central File Header and the Data Descriptor or between the declared value
12   * and the computed value from the decompressed data.
13   * The prior case is detected on the call to {@link ZipFile#getCheckedInputStream},
14   * whereas the latter case is detected when the input stream returned by this
15   * method gets closed.
16   * <p>
17   * The exception's detail message is the name of the ZIP entry.
18   *
19   * @author  Christian Schlichtherle
20   */
21  public class Crc32Exception extends ZipException {
22      private static final long serialVersionUID = 1656298435298526391L;
23  
24      private final long expected, actual;
25  
26      /**
27       * Creates a new instance of {@code Crc32Exception} where the
28       * given entry name forms part of the detail message.
29       *
30       * @param name the entry name.
31       * @param expected the declared CRC32 value.
32       * @param actual the actual CRC32 value;
33       * @see   #getExpectedCrc
34       * @see   #getActualCrc
35       */
36      Crc32Exception(final String name, final long expected, final long actual) {
37          super(name
38                  + " (expected CRC32 value 0x"
39                  + Long.toHexString(expected)
40                  + ", but is actually 0x"
41                  + Long.toHexString(actual)
42                  + ")");
43          assert expected != actual;
44          this.expected = expected;
45          this.actual = actual;
46      }
47  
48      /**
49       * Returns the CRC32 value which has been expected for the ZIP entry.
50       */
51      public long getExpectedCrc() {
52          return expected;
53      }
54  
55      /**
56       * Returns the CRC32 value which has actually been found for the ZIP entry.
57       */
58      public long getActualCrc() {
59          return actual;
60      }
61  }
