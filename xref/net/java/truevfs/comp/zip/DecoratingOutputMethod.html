5   package net.java.truevfs.comp.zip;
6   
7   import edu.umd.cs.findbugs.annotations.CreatesObligation;
8   import java.io.IOException;
9   import java.io.OutputStream;
10  import java.util.zip.ZipException;
11  import javax.annotation.concurrent.NotThreadSafe;
12  
13  /**
14   * A decorator for output methods.
15   * <p>
16   * Implementations cannot be thread-safe.
17   *
18   * @author  Christian Schlichtherle
19   */
20  @NotThreadSafe
21  abstract class DecoratingOutputMethod implements OutputMethod {
22      final OutputMethod method;
23  
24      DecoratingOutputMethod(final OutputMethod processor) {
25          assert null != processor;
26          this.method = processor;
27      }
28  
29      @Override
30      public void init(ZipEntry entry) throws ZipException {
31          method.init(entry);
32      }
33  
34      @Override
35      @CreatesObligation
36      public OutputStream start() throws IOException {
37          return method.start();
38      }
39  
40      @Override
41      public void finish() throws IOException {
42          method.finish();
43      }
44  }
