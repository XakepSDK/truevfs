5   package net.java.truevfs.comp.zip;
6   
7   /**
8    * Provides parameters for reading ZIP files.
9    * <p>
10   * <b>Warning:</b> This interface is <em>not</em> intended for public use
11   * - its API may change at will without prior notification!
12   *
13   * @param  <E> The type of the created ZIP entries.
14   * @author Christian Schlichtherle
15   */
16  public interface ZipFileParameters<E extends ZipEntry>
17  extends ZipCharsetParameters, ZipEntryFactory<E> {
18  
19      /**
20       * Returns the flag for supporting preambles.
21       * If this method returns {@code true}, then a ZIP file is allowed to
22       * contain arbitrary data as its preamble before the actual ZIP file data.
23       * Self Extracting Archives typically use a preamble to store the
24       * application code that is required to extract the ZIP file contents.
25       * <p>
26       * If this method returns {@code false}, then a ZIP file must start with
27       * either a Local File Header (LFH) signature,
28       * a ZIP64 End Of Central Directory Record (EOCDR) signature or an End Of
29       * Central Directory Record (EOCDR) signature.
30       *
31       * @return The flag for allowing a preamble.
32       */
33      boolean getPreambled();
34  
35      /**
36       * Returns the flag for supporting postambles of arbitrary length.
37       * If this method returns {@code true}, then a ZIP file is allowed to
38       * contain arbitrary data of arbitrary length as its postamble after the
39       * actual ZIP file data.
40       * Note that searching for an arbitrary length postamble can seriously
41       * degrade the performance when reading a false positive ZIP file, i.e.
42       * an arbitrary file which is not compatible to the ZIP File Format
43       * Specification, because then the entire file needs to get searched for
44       * an End Of Central Directory Record (EOCDR) signature.
45       * A valid use case are self extracting ZIP files with very large
46       * postambles to host the extraction code.
47       * <p>
48       * If this method returns {@code false}, then a ZIP file may still have a
49       * postamble, but it must not exceed 64KB size, including the End Of
50       * Central Directory record with the ZIP file comment.
51       * This causes the reading of a false positive ZIP file to fail fast.
52       *
53       * @return The flag for supporting postambles of arbitrary length.
54       */
55      boolean getPostambled();
56  }
