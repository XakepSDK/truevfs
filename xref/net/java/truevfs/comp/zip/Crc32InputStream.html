5   package net.java.truevfs.comp.zip;
6   
7   import java.io.IOException;
8   import java.io.InputStream;
9   import java.util.zip.CRC32;
10  
11  /**
12   * Compares the CRC computed from the content to the CRC in the ZIP entry
13   * and throws a Crc32Exception if it detects a mismatch in its method
14   * {@link #close()}.
15   *
16   * @author Christian Schlichtherle
17   */
18  final class Crc32InputStream extends CheckedInputStreamWithSkipBufferSize {
19  
20      private final ZipEntry entry;
21      private boolean closed;
22  
23      Crc32InputStream(
24              final InputStream in,
25              final int skipBufferSize,
26              final ZipEntry entry) {
27          super(in, new CRC32(), skipBufferSize);
28          this.entry = entry;
29      }
30  
31      @Override
32      public void close() throws IOException {
33          if (!closed) {
34              // process CRC-32 until EOF.
35              while (skip(Long.MAX_VALUE) > 0) {
36              }
37          }
38          super.close();
39          closed = true;
40          final long expected = entry.getCrc();
41          final long computed = getChecksum().getValue();
42          if (expected != computed)
43              throw new Crc32Exception(entry.getName(), expected, computed);
44      }
45  }
