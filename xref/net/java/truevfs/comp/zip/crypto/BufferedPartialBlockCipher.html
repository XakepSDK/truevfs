12   * {@link #doFinal(byte[], int)}.

14   * @author Christian Schlichtherle
15   */
16  public final class BufferedPartialBlockCipher extends BufferedBlockCipher {
17  
18      public BufferedPartialBlockCipher(BlockCipher cipher) {
19          super(cipher);
20          partialBlockOkay = true;
21      }
22  }
