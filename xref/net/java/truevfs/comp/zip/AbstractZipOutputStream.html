82      /**
83       * Constructs a raw ZIP output stream which decorates the given output
84       * stream and optionally apppends to the given raw ZIP file.
85       *
86       * @param  sink the sink to write the ZIP file to.
87       *         If {@code appendee} is not {@code null}, then this must be set
88       *         up so that it appends to the same ZIP file from which
89       *         {@code appendee} is reading.
90       * @param  appendee the nullable raw ZIP file to append to.
91       *         This may already be closed.
92       * @param  param the parameters for writing the ZIP file.
93       */
94      @CreatesObligation
95      protected AbstractZipOutputStream(
96              final Sink sink,
97              final @CheckForNull @WillNotClose AbstractZipFile<E> appendee,
98              final ZipOutputStreamParameters param)
99      throws IOException {
100         final OutputStream out = sink.stream();
101         try {
102             this.out = this.leos = null != appendee
103                     ? new AppendingLittleEndianOutputStream(out, appendee)
104                     : new LittleEndianOutputStream(out);
105             if (null != appendee) {
106                 this.charset = appendee.getRawCharset();
107                 this.comment = appendee.getRawComment();
108                 final Map<String, E> entries = new LinkedHashMap<>(
109                         HashMaps.initialCapacity(appendee.size() + param.getOverheadSize()));
110                 entries.putAll(appendee.getRawEntries());
111                 this.entries = entries;
112             } else {
113                 this.charset = param.getCharset();
114                 this.entries = new LinkedHashMap<>(
115                         HashMaps.initialCapacity(param.getOverheadSize()));
116             }
117             setMethod0(param.getMethod());
118             setLevel0(param.getLevel());
119         } catch (final Throwable ex) {
120             try {
121                 out.close();
122             } catch (final Throwable ex2) {
123                 ex.addSuppressed(ex2);
124             }
125             throw ex;
126         }
127     }

312     /**
313      * Starts writing the next ZIP entry to the underlying stream.
314      * Note that if two or more entries with the same name are written
315      * consecutively to this stream, the last entry written will shadow
316      * all other entries, i.e. all of them are written to the ZIP file
317      * (and hence require space), but only the last will be listed in the
318      * central directory.
319      * This is unlike the genuine {@link java.util.zip.ZipOutputStream
320      * java.util.zip.ZipOutputStream} which would throw a {@link ZipException}
321      * in this method when another entry with the same name is to be written.
322      *
323      * @param  entry The entry to write.
324      * @param  process Whether or not the entry contents should get processed,
325      *         e.g. deflated.
326      *         This should be set to {@code false} if and only if the
327      *         application is going to copy entries from an input ZIP file to
328      *         an output ZIP file.
329      *         The entries' CRC-32, compressed size and uncompressed
330      *         size properties must be set in advance.
331      * @throws ZipException If and only if writing the entry is impossible
332      *         because the resulting file would not comply to the ZIP file
333      *         format specification.
334      * @throws IOException On any I/O error.
335      */
336     public void putNextEntry(final E entry, final boolean process)
337     throws ZipException, IOException {
338         closeEntry();
339         final OutputMethod method = newOutputMethod(entry, process);
340         method.init(entry.clone()); // test!
341         method.init(entry);
342         this.out = method.start();
343         this.processor = method;
344         // Store entry now so that a subsequent call to entry(...) returns
345         // it.
346         this.entries.put(entry.getName(), entry);
347         this.entry = entry;
348     }

350     /**
351      * Returns a new output method for the given entry.
352      * Except the property &quot;method&quot;, this method must not modify the
353      * given entry.
354      */
355     @SuppressWarnings("unchecked")
356     private OutputMethod newOutputMethod(
357             final ZipEntry entry,
358             final boolean process)
359     throws ZipException {
360         // HC SVNT DRACONES!
361         OutputMethod processor = new RawOutputMethod(process);
362         if (!process) {
363             assert UNKNOWN != entry.getCrc();
364             return processor;
365         }
366         int method = entry.getMethod();
367         if (UNKNOWN == method)
368             entry.setRawMethod(method = getMethod());
369         boolean skipCrc = false;
370         if (entry.isEncrypted() || WINZIP_AES == method) {
371             ZipCryptoParameters param = getCryptoParameters();
372             if (WINZIP_AES == method) {
373                 param = parameters(WinZipAesParameters.class, param);
374                 final WinZipAesExtraField field = (WinZipAesExtraField)
375                         entry.getExtraField(WINZIP_AES_ID);
376                 if (null != field) {
377                     method = field.getMethod();
378                     if (VV_AE_2 == field.getVendorVersion())
379                         skipCrc = true;
380                 }
381             }
382             processor = newEncryptedOutputMethod((RawOutputMethod) processor,
383                     param);
384         }
385         switch (method) {
386             case STORED:
387                 if (!skipCrc)
388                     processor = new Crc32CheckingOutputMethod(processor);
389                 break;
390             case DEFLATED:
391                 processor = new DeflaterOutputMethod(processor);
392                 if (!skipCrc)
393                     processor = new Crc32UpdatingOutputMethod(processor);
394                 break;
395             case BZIP2:
396                 processor = new BZip2OutputMethod(processor);
397                 if (!skipCrc)
398                     processor = new Crc32UpdatingOutputMethod(processor);
399                 break;
400             default:
401                 throw new ZipException(entry.getName()
402                         + " (unsupported compression method "
403                         + method
404                         + ")");
405         }
406         return processor;
407     }

409     /**
410      * Returns a new {@code EncryptedOutputMethod}.
411      *
412      * @param  processor the output method to decorate.
413      * @param  param the {@link ZipCryptoParameters} used to determine and
414      *         configure the type of the encrypted ZIP file.
415      *         If the run time class of this parameter matches multiple
416      *         parameter interfaces, it is at the discretion of this
417      *         implementation which one is picked and hence which type of
418      *         encrypted ZIP file is created.
419      *         If you need more control over this, pass in an instance which's
420      *         run time class just implements the
421      *         {@link ZipParametersProvider} interface.
422      *         Instances of this interface are queried to find crypto
423      *         parameters which match a known encrypted ZIP file type.
424      *         This algorithm is recursively applied.
425      * @return A new {@code EncryptedOutputMethod}.
426      * @throws ZipParametersException if {@code param} is {@code null} or
427      *         no suitable crypto parameters can get found.
428      */
429     private EncryptedOutputMethod newEncryptedOutputMethod(
430             final RawOutputMethod processor,
431             @CheckForNull ZipParameters param)
432     throws ZipParametersException {
433         assert null != processor;
434         while (null != param) {
435             // Order is important here to support multiple interface implementations!
436             if (param instanceof WinZipAesParameters) {
437                 return new WinZipAesOutputMethod(processor,
438                         (WinZipAesParameters) param);
439             } else if (param instanceof ZipParametersProvider) {
440                 param = ((ZipParametersProvider) param)
441                         .get(ZipCryptoParameters.class);
442             } else {
443                 break;
444             }
445         }
446         throw new ZipParametersException("No suitable crypto parameters available!");
447     }

449     /**
450      * Writes all necessary data for this entry to the underlying stream.
451      *
452      * @throws ZipException If and only if writing the entry is impossible
453      *         because the resulting file would not comply to the ZIP file
454      *         format specification.
455      * @throws IOException On any I/O error.
456      */
457     public void closeEntry() throws IOException {
458         final ZipEntry entry = this.entry;
459         if (null == entry)
460             return;
461         this.processor.finish();
462         this.out.flush();
463         this.out = this.leos;
464         this.processor = null;
465         this.entry = null;
466     }

468     /**
469      * Closes the current entry and writes the Central Directory to the
470      * underlying output stream.
471      * <p>
472      * <b>Notes:</b>
473      * <ul>
474      * <li>The underlying stream is not closed.</li>
475      * <li>Unlike Sun's implementation in J2SE 1.4.2, you may continue to use
476      *     this ZIP output stream with putNextEntry(...) and the like.
477      *     When you finally close the stream, the central directory will
478      *     contain <em>all</em> entries written.</li>
479      * </ul>
480      *
481      * @throws ZipException If and only if writing the entry is impossible
482      *         because the resulting file would not comply to the ZIP file
483      *         format specification.
484      * @throws IOException On any I/O error.
485      */
486     public void finish() throws IOException {
487         if (this.finished)
488             return;
489         closeEntry();
490         final LittleEndianOutputStream leos = this.leos;
491         this.cdOffset = leos.size();
492         final Iterator<E> i = this.entries.values().iterator();
493         while (i.hasNext())
494             if (!writeCentralFileHeader(i.next()))
495                 i.remove();
496         writeEndOfCentralDirectory();
497         this.finished = true;
498     }

500     /**
501      * Writes a Central File Header record.
502      *
503      * @return {@code false} if and only if the record has been skipped,
504      *         i.e. not written for some other reason than an I/O error.
505      * @throws IOException On any I/O error.
506      */
507     private boolean writeCentralFileHeader(final ZipEntry entry)
508     throws IOException {
509         final long csize = entry.getCompressedSize();
510         final long size = entry.getSize();
511         // This test MUST NOT include the CRC-32 because VV_AE_2 sets it to
512         // UNKNOWN!
513         if (UNKNOWN == (csize | size)) {
514             // See http://java.net/jira/browse/TRUEZIP-144 :
515             // The kernel may set any of these properties to UNKNOWN after the
516             // entry content has already been written in order to signal that
517             // this entry should not get included in the central directory.
518             // E.g. this may happen with the GROW output option preference.
519             return false;
520         }
521         final LittleEndianOutputStream leos = this.leos;
522         // central file header signature   4 bytes  (0x02014b50)
523         leos.writeInt(CFH_SIG);
524         // version made by                 2 bytes
525         leos.writeShort((entry.getRawPlatform() << 8) | 63);
526         // version needed to extract       2 bytes
527         leos.writeShort(entry.getRawVersionNeededToExtract());
528         // general purpose bit flag        2 bytes
529         leos.writeShort(entry.getGeneralPurposeBitFlags());
530         // compression method              2 bytes
531         leos.writeShort(entry.getRawMethod());
532         // last mod file time              2 bytes
533         // last mod file date              2 bytes
534         leos.writeInt((int) entry.getRawTime());
535         // crc-32                          4 bytes
536         leos.writeInt((int) entry.getRawCrc());
537         // compressed size                 4 bytes
538         leos.writeInt((int) entry.getRawCompressedSize());
539         // uncompressed size               4 bytes
540         leos.writeInt((int) entry.getRawSize());
541         // file name length                2 bytes
542         final byte[] name = encode(entry.getName());
543         leos.writeShort(name.length);
544         // extra field length              2 bytes
545         final byte[] extra = entry.getRawExtraFields();
546         leos.writeShort(extra.length);
547         // file comment length             2 bytes
548         final byte[] comment = getCommentEncoded(entry);
549         leos.writeShort(comment.length);
550         // disk number start               2 bytes
551         leos.writeShort(0);
552         // internal file attributes        2 bytes
553         leos.writeShort(0);
554         // external file attributes        4 bytes
555         leos.writeInt((int) entry.getRawExternalAttributes());
556         // relative offset of local header 4 bytes
557         leos.writeInt((int) entry.getRawOffset());
558         // file name (variable size)
559         leos.write(name);
560         // extra field (variable size)
561         leos.write(extra);
562         // file comment (variable size)
563         leos.write(comment);
564 
565         return true;
566     }

568     private byte[] getCommentEncoded(final ZipEntry entry) {

572     /**
573      * Writes the End Of Central Directory record.
574      *
575      * @throws IOException On any I/O error.
576      */
577     private void writeEndOfCentralDirectory() throws IOException {
578         final LittleEndianOutputStream leos = this.leos;
579         final long cdEntries = entries.size();
580         final long cdOffset = this.cdOffset;
581         final long cdSize = leos.size() - cdOffset;
582         final boolean cdEntriesZip64 = cdEntries > UShort.MAX_VALUE || FORCE_ZIP64_EXT;
583         final boolean cdSizeZip64    = cdSize    > UInt  .MAX_VALUE || FORCE_ZIP64_EXT;
584         final boolean cdOffsetZip64  = cdOffset  > UInt  .MAX_VALUE || FORCE_ZIP64_EXT;
585         final int cdEntries16 = cdEntriesZip64 ? UShort.MAX_VALUE : (int) cdEntries;
586         final long cdSize32   = cdSizeZip64    ? UInt  .MAX_VALUE : cdSize;
587         final long cdOffset32 = cdOffsetZip64  ? UInt  .MAX_VALUE : cdOffset;
588         final boolean zip64 // ZIP64 extensions?
589                 =  cdEntriesZip64
590                 || cdSizeZip64
591                 || cdOffsetZip64;
592         if (zip64) {
593             final long zip64eocdOffset // relative offset of the zip64 end of central directory record
594                     = leos.size();
595             // zip64 end of central dir
596             // signature                       4 bytes  (0x06064b50)
597             leos.writeInt(ZIP64_EOCDR_SIG);
598             // size of zip64 end of central
599             // directory record                8 bytes
600             leos.writeLong(ZIP64_EOCDR_MIN_LEN - 12);
601             // version made by                 2 bytes
602             leos.writeShort(63);
603             // version needed to extract       2 bytes
604             leos.writeShort(46); // due to potential use of BZIP2 compression
605             // number of this disk             4 bytes
606             leos.writeInt(0);
607             // number of the disk with the
608             // start of the central directory  4 bytes
609             leos.writeInt(0);
610             // total number of entries in the
611             // central directory on this disk  8 bytes
612             leos.writeLong(cdEntries);
613             // total number of entries in the
614             // central directory               8 bytes
615             leos.writeLong(cdEntries);
616             // size of the central directory   8 bytes
617             leos.writeLong(cdSize);
618             // offset of start of central
619             // directory with respect to
620             // the starting disk number        8 bytes
621             leos.writeLong(cdOffset);
622             // zip64 extensible data sector    (variable size)
623             //
624             // zip64 end of central dir locator
625             // signature                       4 bytes  (0x07064b50)
626             leos.writeInt(ZIP64_EOCDL_SIG);
627             // number of the disk with the
628             // start of the zip64 end of
629             // central directory               4 bytes
630             leos.writeInt(0);
631             // relative offset of the zip64
632             // end of central directory record 8 bytes
633             leos.writeLong(zip64eocdOffset);
634             // total number of disks           4 bytes
635             leos.writeInt(1);
636         }
637         // end of central dir signature    4 bytes  (0x06054b50)
638         leos.writeInt(EOCDR_SIG);
639         // number of this disk             2 bytes
640         leos.writeShort(0);
641         // number of the disk with the
642         // start of the central directory  2 bytes
643         leos.writeShort(0);
644         // total number of entries in the
645         // central directory on this disk  2 bytes
646         leos.writeShort(cdEntries16);
647         // total number of entries in
648         // the central directory           2 bytes
649         leos.writeShort(cdEntries16);
650         // size of the central directory   4 bytes
651         leos.writeInt((int) cdSize32);
652         // offset of start of central
653         // directory with respect to
654         // the starting disk number        4 bytes
655         leos.writeInt((int) cdOffset32);
656         // .ZIP file comment length        2 bytes
657         final byte[] comment = getRawComment();
658         leos.writeShort(comment.length);
659         // .ZIP file comment       (variable size)
660         leos.write(comment);
661     }

693     private final class RawOutputMethod implements OutputMethod {
694         final boolean process;
695 
696         /** Start of entry data. */
697         private long dataStart;
698         @Nullable ZipEntry entry;
699 
700         RawOutputMethod(final boolean process) {
701             this.process = process;
702         }
703 
704         @Override
705         public void init(final ZipEntry entry) throws ZipException {
706             {
707                 final long size = encode(entry.getName()).length
708                                 + entry.getRawExtraFields().length
709                                 + encode(entry.getRawComment()).length;
710                 if (UShort.MAX_VALUE < size)
711                     throw new ZipException(entry.getName()
712                             + " (the total size of "
713                             + size
714                             + " bytes for the name, extra fields and comment exceeds the maximum size of "
715                             + UShort.MAX_VALUE + " bytes)");
716             }
717             if (STORED == entry.getMethod() || !this.process) {
718                 if (UNKNOWN == entry.getCrc())
719                     throw new ZipException(entry.getName()
720                             + " (unknown CRC-32 value)");
721                 if (UNKNOWN == entry.getCompressedSize())
722                     throw new ZipException(entry.getName()
723                             + " (unknown compressed size)");
724                 if (UNKNOWN == entry.getSize())
725                     throw new ZipException(entry.getName()
726                             + " (unknown uncompressed size)");
727             }
728             if (UNKNOWN == entry.getPlatform())
729                 entry.setRawPlatform(PLATFORM_FAT);
730             if (UNKNOWN == entry.getTime())
731                 entry.setTime(System.currentTimeMillis());
732             this.entry = entry;
733         }
734 
735         /**
736          * Writes the Local File Header.
737          */
738         @Override
739         public OutputStream start() throws IOException {
740             final LittleEndianOutputStream leos = AbstractZipOutputStream.this.leos;
741             final long offset = leos.size();
742             final ZipEntry entry = this.entry;
743             final boolean encrypted = entry.isEncrypted();
744             final boolean dd = entry.isDataDescriptorRequired();
745             // Compose General Purpose Bit Flag.
746             // See appendix D of PKWARE's ZIP File Format Specification.
747             final boolean utf8 = UTF8.equals(charset);
748             final int general = (encrypted ? GPBF_ENCRYPTED : 0)
749                               | (dd        ? GPBF_DATA_DESCRIPTOR : 0)
750                               | (utf8      ? GPBF_UTF8 : 0);
751             // Start changes.
752             AbstractZipOutputStream.this.finished = false;
753             // local file header signature     4 bytes  (0x04034b50)
754             leos.writeInt(LFH_SIG);
755             // version needed to extract       2 bytes
756             leos.writeShort(entry.getRawVersionNeededToExtract());
757             // general purpose bit flag        2 bytes
758             leos.writeShort(general);
759             // compression method              2 bytes
760             leos.writeShort(entry.getRawMethod());
761             // last mod file time              2 bytes
762             // last mod file date              2 bytes
763             leos.writeInt((int) entry.getRawTime());
764             // crc-32                          4 bytes
765             // compressed size                 4 bytes
766             // uncompressed size               4 bytes
767             if (dd) {
768                 leos.writeInt(0);
769                 leos.writeInt(0);
770                 leos.writeInt(0);
771             } else {
772                 leos.writeInt((int) entry.getRawCrc());
773                 leos.writeInt((int) entry.getRawCompressedSize());
774                 leos.writeInt((int) entry.getRawSize());
775             }
776             // file name length                2 bytes
777             final byte[] name = encode(entry.getName());
778             leos.writeShort(name.length);
779             // extra field length              2 bytes
780             final byte[] extra = entry.getRawExtraFields();
781             leos.writeShort(extra.length);
782             // file name (variable size)
783             leos.write(name);
784             // extra field (variable size)
785             leos.write(extra);
786             // Commit changes.
787             entry.setGeneralPurposeBitFlags(general);
788             entry.setRawOffset(offset);
789             // Update data start.
790             this.dataStart = leos.size();
791             return leos;
792         }
793 
794         /**
795          * Checks the compressed entry size and optionally writes the Data
796          * Descriptor.
797          */
798         @Override
799         public void finish() throws IOException {
800             final LittleEndianOutputStream leos = AbstractZipOutputStream.this.leos;
801             final long csize = leos.size() - this.dataStart;
802             final ZipEntry entry = this.entry;
803             assert UNKNOWN != entry.getCrc();
804             assert UNKNOWN != entry.getSize();
805             if (entry.getGeneralPurposeBitFlag(GPBF_DATA_DESCRIPTOR)) {
806                 entry.setRawCompressedSize(csize);
807                 // data descriptor signature       4 bytes  (0x08074b50)
808                 leos.writeInt(DD_SIG);
809                 // crc-32                          4 bytes
810                 leos.writeInt((int) entry.getRawCrc());
811                 // compressed size                 4 or 8 bytes
812                 // uncompressed size               4 or 8 bytes
813                 if (entry.isZip64ExtensionsRequired()) {
814                     leos.writeLong(csize);
815                     leos.writeLong(entry.getSize());
816                 } else {
817                     leos.writeInt((int) entry.getRawCompressedSize());
818                     leos.writeInt((int) entry.getRawSize());
819                 }
820             } else if (entry.getCompressedSize() != csize) {
821                 throw new ZipException(entry.getName()
822                         + " (expected compressed entry size of "
823                         + entry.getCompressedSize()
824                         + " bytes, but is actually "
825                         + csize
826                         + " bytes)");
827             }
828         }
829     } // RawOutputMethod

837     private final class WinZipAesOutputMethod extends EncryptedOutputMethod {
838         final WinZipAesParameters generalParam;
839         boolean suppressCrc;
840         @Nullable WinZipAesEntryParameters entryParam;
841         @Nullable WinZipAesOutputStream out;
842         @Nullable ZipEntry entry;
843 
844         WinZipAesOutputMethod(
845                 RawOutputMethod processor,
846                 final WinZipAesParameters param) {
847             super(processor);
848             assert null != param;
849             this.generalParam = param;
850         }
851 
852         @Override
853         public void init(final ZipEntry entry) throws ZipException {
854             // HC SVNT DRACONES!
855             final WinZipAesEntryParameters entryParam
856                     = new WinZipAesEntryParameters(this.generalParam, entry);
857             final AesKeyStrength keyStrength = entryParam.getKeyStrength();
858             this.entryParam = entryParam;
859             WinZipAesExtraField field = null;
860             int method = entry.getMethod();
861             long csize = entry.getCompressedSize();
862             if (WINZIP_AES == method) {
863                 field = (WinZipAesExtraField) entry.getExtraField(
864                         WINZIP_AES_ID);
865                 if (null != field) {
866                     method = field.getMethod();
867                     if (UNKNOWN != csize)
868                         csize -= overhead(field.getKeyStrength());
869                     entry.setRawMethod(method); // restore for out.init(*)
870                 }
871             }
872             if (null == field)
873                 field = new WinZipAesExtraField();
874             field.setKeyStrength(keyStrength);
875             field.setMethod(method);
876             final long size = entry.getSize();
877             if (20 <= size && BZIP2 != method) {
878                 field.setVendorVersion(VV_AE_1);
879             } else {
880                 field.setVendorVersion(VV_AE_2);
881                 this.suppressCrc = true;
882             }
883             entry.addExtraField(field);
884             if (UNKNOWN != csize) {
885                 csize += overhead(keyStrength);
886                 entry.setRawCompressedSize(csize);
887             }
888             if (this.suppressCrc) {
889                 final long crc = entry.getCrc();
890                 entry.setRawCrc(0);
891                 this.method.init(entry);
892                 entry.setCrc(crc);
893             } else {
894                 this.method.init(entry);
895             }
896             entry.setRawMethod(WINZIP_AES);
897             this.entry = entry;
898         }
899 
900         @Override
901         public OutputStream start() throws IOException {
902             // see DeflatedOutputMethod.finish().
903             final ZipEntry entry = this.entry;
904             final OutputMethod method = this.method;
905             final WinZipAesEntryParameters entryParam = this.entryParam;
906             assert null != entryParam;
907             assert null == this.out;
908             if (suppressCrc) {
909                 final long crc = entry.getCrc();
910                 entry.setRawCrc(0);
911                 this.out = new WinZipAesOutputStream(entryParam,
912                         (LittleEndianOutputStream) method.start());
913                 entry.setCrc(crc);
914             } else {
915                 this.out = new WinZipAesOutputStream(entryParam,
916                         (LittleEndianOutputStream) method.start());
917             }
918             return this.out;
919         }
920 
921         @Override
922         public void finish() throws IOException {
923             // see DeflatedOutputMethod.finish().
924             assert null != this.out;
925             this.out.finish();
926             if (this.suppressCrc) {
927                 final ZipEntry entry = this.entry;
928                 entry.setRawCrc(0);
929                 this.method.finish();
930                 // Set to UNKNOWN in order to signal to
931                 // Crc32CheckingOutputMethod that it should not check it and
932                 // signal to writeCentralFileHeader() that it should write 0.
933                 entry.setCrc(UNKNOWN);
934             } else {
935                 this.method.finish();
936             }
937         }

940     private final class BZip2OutputMethod extends DecoratingOutputMethod {
941         @Nullable BZip2CompressorOutputStream cout;
942         @Nullable LittleEndianOutputStream dout;
943         @Nullable ZipEntry entry;
944 
945         BZip2OutputMethod(OutputMethod processor) {
946             super(processor);
947         }
948 
949         @Override
950         public void init(final ZipEntry entry) throws ZipException  {
951             entry.setCompressedSize(UNKNOWN);
952             this.method.init(entry);
953             this.entry = entry;
954         }
955 
956         @Override
957         public OutputStream start() throws IOException {
958             assert null == this.cout;
959             assert null == this.dout;
960             OutputStream out = this.method.start();
961             final long size = this.entry.getSize();
962             final int blockSize = UNKNOWN != size
963                     ? BZip2CompressorOutputStream.chooseBlockSize(size)
964                     : getBZip2BlockSize();
965             out = this.cout = new BZip2CompressorOutputStream(out, blockSize);
966             return this.dout = new LittleEndianOutputStream(out);
967         }
968 
969         int getBZip2BlockSize() {
970             final int level = AbstractZipOutputStream.this.getLevel();
971             if (BZip2CompressorOutputStream.MIN_BLOCKSIZE <= level
972                     && level <= BZip2CompressorOutputStream.MAX_BLOCKSIZE)
973                 return level;
974             return BZip2CompressorOutputStream.MAX_BLOCKSIZE;
975         }
976 
977         @Override
978         public void finish()
979         throws IOException {
980             this.dout.flush(); // superfluous - should not buffer
981             this.cout.finish();
982             this.entry.setRawSize(this.dout.size());
983             this.method.finish();
984         }
