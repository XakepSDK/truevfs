5   package net.java.truevfs.comp.zip;
6   
7   import javax.annotation.CheckForNull;
8   import javax.annotation.concurrent.ThreadSafe;
9   
10  /**
11   * Provides static utility methods for ZIP parameters.
12   * 
13   * @author Christian Schlichtherle
14   */
15  @ThreadSafe
16  final class ZipParametersUtils {
17  
18      private ZipParametersUtils() {
19      }
20  
21      /**
22       * Searches for {@code ZipParameters} of the given type starting from
23       * the given parameters.
24       * 
25       * @param  <P> the type of the ZIP parameters to search for.
26       * @param  type the type of the ZIP parameters to search for.
27       * @param  param the parameters for starting the search.
28       * @return The parameters of the given type.
29       * @throws ZipParametersException if {@code param} is {@code null} or
30       *         no suitable parameters can get found.
31       */
32      static <P extends ZipParameters> P parameters(
33              final Class<P> type,
34              @CheckForNull ZipParameters param)
35      throws ZipParametersException {
36          while (null != param) {
37              // Order is important here to support multiple interface implementations!
38              if (type.isInstance(param)) {
39                  return type.cast(param);
40              } else if (param instanceof ZipParametersProvider) {
41                  param = ((ZipParametersProvider) param).get(type);
42              } else {
43                  break;
44              }
45          }
46          throw new ZipParametersException("No suitable ZIP parameters available!");
47      }
48  }
