38  /**
39   * Provides unsafe (raw) access to a ZIP file using shared {@link ZipEntry}
40   * instances.
41   * <p>
42   * <b>Warning:</b> This class is <em>not</em> intended for public use
43   * - its API may change at will without prior notification!
44   * <p>
45   * Where the constructors of this class accept a {@code charset}
46   * parameter, this is used to decode comments and entry names in the ZIP file.
47   * However, if an entry has bit 11 set in its General Purpose Bit Flags,
48   * then this parameter is ignored and "UTF-8" is used for this entry.
49   * This is in accordance to Appendix D of PKWARE's
50   * <a href="http://www.pkware.com/documents/casestudies/APPNOTE.TXT">ZIP File Format Specification</a>,
51   * version 6.3.0 and later.
52   * <p>
53   * This class is able to skip a preamble like the one found in self extracting
54   * archives.
55   *
56   * @param  <E> the type of the ZIP entries.
57   * @see    AbstractZipOutputStream
58   * @author Christian Schlichtherle
59   */
60  @NotThreadSafe
61  @CleanupObligation
62  public abstract class AbstractZipFile<E extends ZipEntry>
63  implements Closeable, Iterable<E> {

66              /* Local File Header signature     */ 4 +
67              /* Version Needed To Extract       */ 2 +
68              /* General Purpose Bit Flags       */ 2 +
69              /* Compression Method              */ 2 +
70              /* Last Mod File Time              */ 2 +
71              /* Last Mod File Date              */ 2 +
72              /* CRC-32                          */ 4 +
73              /* Compressed Size                 */ 4 +
74              /* Uncompressed Size               */ 4;

113     /**
114      * Reads the given {@code zip} file in order to provide random access
115      * to its entries.
116      *
117      * @param  source the source for reading the ZIP file from.
118      * @param  param the parameters for reading the ZIP file.
119      * @throws ZipException if the source data is not compatible to the ZIP
120      *         File Format Specification.
121      * @throws EOFException on unexpected end-of-file.
122      * @throws IOException on any I/O error.
123      * @see    #recoverLostEntries()
124      */
125     @CreatesObligation
126     protected AbstractZipFile(
127             final Source source,
128             final ZipFileParameters<E> param)
129     throws ZipException, EOFException, IOException {
130         this.param = param;
131         final SeekableByteChannel channel = this.channel = source.channel();
132         try {
133             length = channel.size();
134             charset = param.getCharset();
135             final @WillNotClose SeekableByteChannel
136                     bchannel = new SafeBufferedReadOnlyChannel(channel, length);
137             if (!param.getPreambled()) checkZipFileSignature(bchannel);
138             final int numEntries = findCentralDirectory(bchannel, param.getPostambled());
139             mountCentralDirectory(bchannel, numEntries);
140             if (preamble + postamble >= length) {
141                 assert 0 == numEntries;
142                 if (param.getPreambled()) // otherwise already checked
143                     checkZipFileSignature(bchannel);
144             }
145             assert null != channel;
146             assert null != charset;
147             assert null != entries;
148             assert null != mapper;
149             // Do NOT close bchannel - would close channel as well!
150         } catch (final Throwable e1) {
151             try {
152                 channel.close();
153             } catch (final Throwable e2) {
154                 e1.addSuppressed(e2);
155             }
156             throw e1;
157         }
158     }

160     private void checkZipFileSignature(final SeekableByteChannel channel)
161     throws IOException {
162         final long sig = MutableBuffer
163                 .allocate(4)
164                 .littleEndian()
165                 .load(channel.position(preamble))
166                 .getUInt();
167         // Constraint: A ZIP file must start with a Local File Header
168         // or a (ZIP64) End Of Central Directory Record iff it's emtpy.
169         if (LFH_SIG != sig
170                   && ZIP64_EOCDR_SIG != sig
171                   && EOCDR_SIG != sig)
172             throw new ZipException(
173                     "Expected local file header or (ZIP64) end of central directory record!");
174     }

176     /**
177      * Positions the file pointer at the first Central File Header.
178      * Performs some means to check that this is really a ZIP file.
179      * <p>
180      * As a side effect, the following fields will get initialized:
181      * <ul>
182      * <li>{@link #preamble}
183      * <li>{@link #postamble}
184      * </ul>
185      * <p>
186      * The following fields may get updated:
187      * <ul>
188      * <li>{@link #comment}
189      * <li>{@link #mapper}
190      * </ul>
191      *
192      * @throws ZipException If the file is not compatible to the ZIP File
193      *         Format Specification.
194      * @throws IOException On any other I/O error.
195      */
196     private int findCentralDirectory(
197             final SeekableByteChannel channel,
198             final boolean postambled)
199     throws IOException {
200         // Search for End Of Central Directory Record.
201         final MutableBuffer eocdr = MutableBuffer
202                 .allocate(EOCDR_MIN_LEN)
203                 .littleEndian();
204         final long max = length - EOCDR_MIN_LEN;
205         final long min = !postambled && max >= 0xffff ? max - 0xffff : 0;
206         for (long eocdrPos = max; eocdrPos >= min; eocdrPos--) {
207             eocdr.rewind().limit(4).load(channel.position(eocdrPos));
208             // end of central dir signature    4 bytes  (0x06054b50)
209             if (EOCDR_SIG != eocdr.getUInt()) continue;
210             try {
211                 // Process End Of Central Directory Record.
212                 eocdr.limit(EOCDR_MIN_LEN).load(channel);
213                 // number of this disk             2 bytes
214                 long diskNo = eocdr.getUShort();
215                 // number of the disk with the
216                 // start of the central directory  2 bytes
217                 long cdDiskNo = eocdr.getUShort();
218                 // total number of entries in the
219                 // central directory on this disk  2 bytes
220                 long cdEntriesDisk = eocdr.getUShort();
221                 // total number of entries in
222                 // the central directory           2 bytes
223                 long cdEntries = eocdr.getUShort();
224                 if (0 != diskNo || 0 != cdDiskNo || cdEntriesDisk != cdEntries)
225                     throw new ZipException(
226                             "ZIP file spanning/splitting is not supported!");
227                 // size of the central directory   4 bytes
228                 long cdSize = eocdr.getUInt();
229                 // offset of start of central
230                 // directory with respect to
231                 // the starting disk number        4 bytes
232                 long cdPos = eocdr.getUInt();
233                 // .ZIP file comment length        2 bytes
234                 int commentLen = eocdr.getUShort();
235                 // .ZIP file comment       (variable size)
236                 if (0 < commentLen)
237                     comment = MutableBuffer
238                             .allocate(commentLen)
239                             .load(channel)
240                             .array();
241                 preamble = eocdrPos;
242                 postamble = length - channel.position();
243 
244                 // Check for ZIP64 End Of Central Directory Locator.
245                 final long eocdlPos = eocdrPos - ZIP64_EOCDL_LEN;
246                 final MutableBuffer zip64eocdl = MutableBuffer
247                     .allocate(ZIP64_EOCDL_LEN)
248                     .littleEndian();
249                 // zip64 end of central dir locator
250                 // signature                       4 bytes  (0x07064b50)
251                 if (0 > eocdlPos || ZIP64_EOCDL_SIG != zip64eocdl
252                         .load(channel.position(eocdlPos))
253                         .getUInt()) {
254                     // Seek and check first CFH, probably requiring an offset mapper.
255                     long offset = eocdrPos - cdSize;
256                     channel.position(offset);
257                     offset -= cdPos;
258                     if (0 != offset) mapper = new OffsetPositionMapper(offset);
259                     return (int) cdEntries;
260                 }
261 
262                 // number of the disk with the
263                 // start of the zip64 end of
264                 // central directory               4 bytes
265                 final long zip64eocdrDisk = zip64eocdl.getUInt();
266                 // relative offset of the zip64
267                 // end of central directory record 8 bytes
268                 final long zip64eocdrPos = zip64eocdl.getLong();
269                 // total number of disks           4 bytes
270                 final long totalDisks = zip64eocdl.getUInt();
271                 if (0 != zip64eocdrDisk || 1 != totalDisks)
272                     throw new ZipException(
273                             "ZIP file spanning/splitting is not supported!");
274 
275                 // Read Zip64 End Of Central Directory Record.
276                 final MutableBuffer zip64eocdr = MutableBuffer
277                         .allocate(ZIP64_EOCDR_MIN_LEN)
278                         .littleEndian()
279                         .load(channel.position(zip64eocdrPos));
280                 // zip64 end of central dir
281                 // signature                       4 bytes  (0x06064b50)
282                 if (ZIP64_EOCDR_SIG != zip64eocdr.getUInt())
283                     throw new ZipException(
284                             "Expected ZIP64 end of central directory record!");
285                 // size of zip64 end of central
286                 // directory record                8 bytes
287                 // version made by                 2 bytes
288                 // version needed to extract       2 bytes
289                 zip64eocdr.skip(8 + 2 + 2);
290                 // number of this disk             4 bytes
291                 diskNo = zip64eocdr.getUInt();
292                 // number of the disk with the
293                 // start of the central directory  4 bytes
294                 cdDiskNo = zip64eocdr.getUInt();
295                 // total number of entries in the
296                 // central directory on this disk  8 bytes
297                 cdEntriesDisk = zip64eocdr.getLong();
298                 // total number of entries in the
299                 // central directory               8 bytes
300                 cdEntries = zip64eocdr.getLong();
301                 if (0 != diskNo || 0 != cdDiskNo || cdEntriesDisk != cdEntries)
302                     throw new ZipException(
303                             "ZIP file spanning/splitting is not supported!");
304                 if (cdEntries < 0 || Integer.MAX_VALUE < cdEntries)
305                     throw new ZipException(
306                             "Total number of entries in the central directory out of range!");
307                 // size of the central directory   8 bytes
308                 //cdSize = zip64eocdr.getLong();
309                 zip64eocdr.skip(8);
310                 // offset of start of central
311                 // directory with respect to
312                 // the starting disk number        8 bytes
313                 cdPos = zip64eocdr.getLong();
314                 // zip64 extensible data sector    (variable size)
315                 channel.position(cdPos);
316                 preamble = zip64eocdrPos;
317                 return (int) cdEntries;
318             } catch (RuntimeException e) {
319                 throw (ZipException) new ZipException(
320                         "Invalid (ZIP64) End Of Central Directory Record")
321                         .initCause(e);
322             }
323         }
324 
325         // Start recovering file entries from min.
326         preamble = min;
327         postamble = length - min;
328         return 0;
329     }

331     /**
332      * Reads the central directory from the given seekable byte channel and
333      * populates the internal tables with ZipEntry instances.
334      * <p>
335      * The ZipEntrys will know all data that can be obtained from
336      * the central directory alone, but not the data that requires the
337      * local file header or additional data to be read.
338      * <p>
339      * As a side effect, the following fields will get initialized:
340      * <ul>
341      * <li>{@link #entries}
342      * </ul>
343      * <p>
344      * The following fields may get updated:
345      * <ul>
346      * <li>{@link #preamble}
347      * <li>{@link #charset}
348      * </ul>
349      *
350      * @throws ZipException If the file is not compatible to the ZIP File
351      *         Format Specification.
352      * @throws IOException on any I/O error.
353      */
354     private void mountCentralDirectory(
355             final SeekableByteChannel channel,
356             int numEntries)
357     throws IOException {
358         final MutableBuffer cfh = MutableBuffer
359                 .allocate(CFH_MIN_LEN)
360                 .littleEndian();
361         final Map<String, E> entries = new LinkedHashMap<>(
362                 Math.max(HashMaps.initialCapacity(numEntries), 16));
363         for (; ; numEntries--) {
364             cfh.rewind().limit(4).load(channel);
365             // central file header signature   4 bytes  (0x02014b50)
366             if (CFH_SIG != cfh.getUInt()) break;
367             cfh.limit(CFH_MIN_LEN).load(channel);
368             final int gpbf = cfh.position(8).getUShort();
369             final int nameLen = cfh.position(28).getUShort();
370             final MutableBuffer name = MutableBuffer
371                     .allocate(nameLen)
372                     .load(channel);
373             // See appendix D of PKWARE's ZIP File Format Specification.
374             final boolean utf8 = 0 != (gpbf & GPBF_UTF8);
375             if (utf8) charset = UTF8;
376             final E entry = param.newEntry(decode(name.array()));
377             try {
378                 // central file header signature   4 bytes  (0x02014b50)
379                 cfh.position(4);
380                 // version made by                 2 bytes
381                 entry.setRawPlatform(cfh.getUShort() >> 8);
382                 // version needed to extract       2 bytes
383                 // general purpose bit flag        2 bytes
384                 cfh.skip(2 + 2);
385                 entry.setGeneralPurposeBitFlags(gpbf);
386                 // compression method              2 bytes
387                 entry.setRawMethod(cfh.getUShort());
388                 // last mod file time              2 bytes
389                 // last mod file date              2 bytes
390                 entry.setRawTime(cfh.getUInt());
391                 // crc-32                          4 bytes
392                 entry.setRawCrc(cfh.getUInt());
393                 // compressed size                 4 bytes
394                 entry.setRawCompressedSize(cfh.getUInt());
395                 // uncompressed size               4 bytes
396                 entry.setRawSize(cfh.getUInt());
397                 // file name length                2 bytes
398                 cfh.skip(2);
399                 // extra field length              2 bytes
400                 final int extraLen = cfh.getUShort();
401                 // file comment length             2 bytes
402                 final int commentLen = cfh.getUShort();
403                 // disk number start               2 bytes
404                 // internal file attributes        2 bytes
405                 cfh.skip(2 + 2);
406                 //entry.setEncodedInternalAttributes(readUShort(cfh, off));
407                 // external file attributes        4 bytes
408                 entry.setRawExternalAttributes(cfh.getUInt());
409                 // relative offset of local header 4 bytes
410                 long lfhOff = cfh.getUInt();
411                 entry.setRawOffset(lfhOff); // must be unmapped!
412                 // extra field (variable size)
413                 if (0 < extraLen)
414                     entry.setRawExtraFields(MutableBuffer
415                             .allocate(extraLen)
416                             .load(channel)
417                             .array());
418                 // file comment (variable size)
419                 if (0 < commentLen)
420                     entry.setRawComment(decode(MutableBuffer
421                             .allocate(commentLen)
422                             .load(channel)
423                             .array()));
424                 // Re-load virtual offset after ZIP64 Extended Information
425                 // Extra Field may have been parsed, map it to the real
426                 // offset and conditionally update the preamble size from it.
427                 lfhOff = mapper.map(entry.getOffset());
428                 if (lfhOff < preamble) preamble = lfhOff;
429             } catch (RuntimeException e) {
430                 throw (ZipException) new ZipException(
431                         entry.getName() + " (invalid Central File Header)")
432                         .initCause(e);
433             }
434 
435             // Map the entry using the name that has been determined
436             // by the ZipEntryFactory.
437             // Note that this name may differ from what has been found
438             // in the ZIP file!
439             entries.put(entry.getName(), entry);
440         }
441 
442         // Check if the number of entries found matches the number of entries
443         // declared in the (ZIP64) End Of Central Directory header.
444         // Sometimes, legacy ZIP32 archives (those without ZIP64 extensions)
445         // contain more than the maximum number of entries specified in the
446         // ZIP File Format Specification, which is 65535 (= 0xffff, a two byte
447         // unsigned integer).
448         // In this case, the declared number of entries usually overflows and
449         // may get negative (Java does not support unsigned integers).
450         // Although beyond the spec, we silently tolerate this in the test.
451         // Thanks to Jean-Francois Thamie for this hint!
452         if (0 != numEntries % 0x10000)
453             throw new ZipException(
454                     "Expected " +
455                     Math.abs(numEntries) +
456                     (numEntries > 0 ? " more" : " less") +
457                     " entries in the central directory!");
458 
459         // Commit map of entries.
460         this.entries = entries;
461     }

463     /**
464      * Recovers any lost entries which have been added to the ZIP file after
465      * the (last) End Of Central Directory Record (EOCDR).
466      * This method should be called immediately after the constructor.
467      * It requires a fully initialized object, hence it's not part of the
468      * constructor.
469      * For example, to recover encrypted entries, it may require
470      * {@link #getCryptoParameters() crypto parameters}.
471      * <p>
472      * This method starts parsing entries at the start of the postamble and
473      * continues until it hits EOF or any non-entry data.
474      * As a side effect, it will not only add any found entries to its internal
475      * map, but will also cut the start of the postamble accordingly.
476      * <p>
477      * Note that it's very likely that this method terminates with an
478      * exception unless the postamble is empty or contains only valid ZIP
479      * entries.
480      * Therefore it may be a good idea to log or silently ignore any exception
481      * thrown by this method.
482      * If an exception is thrown, you can check and recover the remaining
483      * postamble for post-mortem analysis by calling
484      * {@link #getPostambleLength()} and {@link #getPostambleInputStream()}.
485      *
486      * @return {@code this}
487      * @throws ZipException if an invalid entry is found.
488      * @throws EOFException on unexpected end-of-file.
489      * @throws IOException on any I/O error.
490      */
491     public AbstractZipFile<E> recoverLostEntries()
492     throws ZipException, EOFException, IOException {
493         final long length = this.length;
494         final SeekableByteChannel
495                 channel = new SafeBufferedReadOnlyChannel(channel(), length);
496         while (0 < postamble) {
497             long pos = length - postamble;
498             final MutableBuffer lfh = MutableBuffer
499                     .allocate(LFH_MIN_LEN)
500                     .littleEndian()
501                     .load(channel.position(pos));
502             if (LFH_SIG != lfh.getUInt())
503                 throw new ZipException("Expected local file header!");
504             final int gpbf = lfh.position(6).getUShort();
505             final int nameLen = lfh.position(26).getUShort();
506             // See appendix D of PKWARE's ZIP File Format Specification.
507             if (0 != (gpbf & GPBF_UTF8)) charset = UTF8;
508             final E entry = param.newEntry(decode(MutableBuffer
509                     .allocate(nameLen)
510                     .load(channel)
511                     .array()));
512             try {
513                 // local file header signature     4 bytes  (0x04034b50)
514                 // version needed to extract       2 bytes
515                 // general purpose bit flag        2 bytes
516                 entry.setGeneralPurposeBitFlags(gpbf);
517                 lfh.position(8);
518                 // compression method              2 bytes
519                 entry.setRawMethod(lfh.getUShort());
520                 // last mod file time              2 bytes
521                 // last mod file date              2 bytes
522                 entry.setRawTime(lfh.getUInt());
523                 // crc-32                          4 bytes
524                 entry.setRawCrc(lfh.getUInt());
525                 // compressed size                 4 bytes
526                 entry.setRawCompressedSize(lfh.getUInt());
527                 // uncompressed size               4 bytes
528                 entry.setRawSize(lfh.getUInt());
529                 // file name length                2 bytes
530                 lfh.skip(2);
531                 // extra field length              2 bytes
532                 final int extraLen = lfh.getUShort();
533                 entry.setRawOffset(mapper.unmap(pos));
534                 // extra field (variable size)
535                 if (0 < extraLen)
536                     entry.setRawExtraFields(MutableBuffer
537                             .allocate(extraLen)
538                             .load(channel)
539                             .array());
540 
541                 // Process entry contents.
542                 if (entry.getGeneralPurposeBitFlag(GPBF_DATA_DESCRIPTOR)) {
543                     // HC SVNT DRACONES!
544                     // This is the tough one.
545                     // We need to process the entry as if we were unzipping
546                     // it because the CRC-32, the compressed size and the
547                     // uncompressed size are unknown.
548                     // Once we have done this, we compare our findings to
549                     // the Data Descriptor which comes next.
550                     final long start = pos = channel.position();
551                     SeekableByteChannel echannel = new IntervalReadOnlyChannel(
552                             channel, pos, length - pos);
553                     WinZipAesExtraField field = null;
554                     int method = entry.getMethod();
555                     if (entry.isEncrypted()) {
556                         if (WINZIP_AES != method)
557                             throw new ZipException(entry.getName()
558                                     + " (encrypted compression method "
559                                     + method
560                                     + " is not supported)");
561                         echannel = new WinZipAesReadOnlyChannel(echannel,
562                                 new WinZipAesEntryParameters(
563                                     parameters(
564                                         WinZipAesParameters.class,
565                                         getCryptoParameters()),
566                                     entry));
567                         field = (WinZipAesExtraField)
568                                 entry.getExtraField(WINZIP_AES_ID);
569                         method = field.getMethod();
570                     }
571                     final int bufSize = getBufferSize(entry);
572                     CountingInputStream din = null;
573                     InputStream in;
574                     switch (method) {
575                         case DEFLATED:
576                             in = new ZipInflaterInputStream(
577                                     new DummyByteChannelInputStream(echannel),
578                                     bufSize);
579                             break;
580                         case BZIP2:
581                             din = new CountingInputStream(
582                                     new ChannelInputStream(echannel));
583                             try {
584                                 in = new BZip2CompressorInputStream(din);
585                             } catch (final Throwable e1) {
586                                 try {
587                                     din.close();
588                                 } catch (final Throwable e2) {
589                                     e1.addSuppressed(e2);
590                                 }
591                                 throw e1;
592                             }
593                             break;
594                         default:
595                             throw new ZipException(entry.getName()
596                                     + " (compression method "
597                                     + method
598                                     + " is not supported)");
599                     }
600                     try (final CheckedInputStream cin = new CheckedInputStream(in, new CRC32())) {
601                         entry.setRawSize(cin.skip(Long.MAX_VALUE));
602                         if (null != field && field.getVendorVersion() == VV_AE_2)
603                             entry.setRawCrc(0);
604                         else
605                             entry.setRawCrc(cin.getChecksum().getValue());
606                         // Sync file pointer on deflated input again.
607                         switch (method) {
608                             case DEFLATED:
609                                 final Inflater inf = ((ZipInflaterInputStream) in)
610                                         .getInflater();
611                                 assert inf.finished();
612                                 pos += inf.getBytesRead();
613                                 break;
614                             case BZIP2:
615                                 pos += din.getBytesRead();
616                                 break;
617                             default:
618                                 throw new AssertionError();
619                         }
620                     }
621                     if (null != field) pos += overhead(field.getKeyStrength());
622                     entry.setRawCompressedSize(pos - start);
623 
624                     // We have reconstituted all meta data for the entry.
625                     // Next comes the Data Descriptor.
626                     // Let's parse and check it.
627                     final MutableBuffer dd = MutableBuffer
628                             .allocate(entry.isZip64ExtensionsRequired()
629                                 ? 4 + 8 + 8
630                                 : 4 + 4 + 4)
631                             .littleEndian()
632                             .limit(4)
633                             .load(channel.position(pos));
634                     long crc = dd.getUInt();
635                     // Note the Data Descriptor's Signature is optional:
636                     // All newer apps should write it (and so does TrueVFS),
637                     // but older apps might not.
638                     if (DD_SIG == crc) dd.rewind();
639                     dd.limit(dd.capacity()).load(channel);
640                     crc = dd.position(0).getUInt();
641                     final long csize;
642                     final long size;
643                     if (entry.isZip64ExtensionsRequired()) {
644                         csize = dd.getLong();
645                         size  = dd.getLong();
646                     } else {
647                         csize = dd.getUInt();
648                         size  = dd.getUInt();
649                     }
650                     if (entry.getCrc() != crc)
651                         throw new Crc32Exception(entry.getName(),
652                                 entry.getCrc(), crc);
653                     if (entry.getCompressedSize() != csize)
654                         throw new ZipException(entry.getName()
655                                 + " (invalid compressed size in data descriptor)");
656                     if (entry.getSize() != size)
657                         throw new ZipException(entry.getName()
658                                 + " (invalid uncompressed size in data descriptor)");
659                 } else {
660                     // This is the easy one.
661                     // The entry is not using a Data Descriptor, so we can
662                     // use the properties parsed from the Local File Header.
663                     pos += entry.getCompressedSize();
664                     channel.position(pos - 1);
665                     if (pos > length || channel.position() >= channel.size())
666                         throw new ZipException(entry.getName()
667                                 + " (truncated entry data)");
668                 }
669             } catch (RuntimeException e) {
670                 throw (ZipException) new ZipException(
671                         entry.getName() + " (invalid Local File Header or Data Descriptor)")
672                         .initCause(e);
673             }
674 
675             // Entry is almost recovered. Update the postamble length.
676             postamble = length - channel.position();
677 
678             // Map the entry using the name that has been determined
679             // by the ZipEntryFactory.
680             // Note that this name may differ from what has been found
681             // in the ZIP file!
682             entries.put(entry.getName(), entry);
683         }
684         return this;
685     }

859     /**
860      * Returns an {@code InputStream} for reading the contents of the given
861      * entry.
862      * <p>
863      * If the {@link #close} method is called on this instance, all input
864      * streams returned by this method are closed, too.
865      *
866      * @param  name The name of the entry to get the stream for.
867      * @param  check Whether or not the entry content gets checked/authenticated.
868      *         If the parameter {@code process} is {@code false}, then this
869      *         parameter is ignored.
870      *         Otherwise, if this parameter is {@code null}, then it is set to
871      *         the {@link ZipEntry#isEncrypted()} property of the given entry.
872      *         Finally, if this parameter is {@code true},
873      *         then the following additional check is performed for the entry:
874      *         <ul>
875      *         <li>If the entry is encrypted, then the Message Authentication
876      *             Code (MAC) value gets computed and checked.
877      *             If this check fails, then a
878      *             {@link ZipAuthenticationException} gets thrown from this
879      *             method (pre-check).
880      *         <li>If the entry is <em>not</em> encrypted, then the CRC-32
881      *             value gets computed and checked.
882      *             First, the local file header is checked to hold the same
883      *             CRC-32 value than the central directory record.
884      *             Second, the CRC-32 value is computed and checked.
885      *             If this check fails, then a {@link Crc32Exception}
886      *             gets thrown when {@link InputStream#close} is called on the
887      *             returned entry stream (post-check).
888      *         </ul>
889      * @param  process Whether or not the entry contents should get processed,
890      *         e.g. inflated.
891      *         This should be set to {@code false} if and only if the
892      *         application is going to copy entries from an input ZIP file to
893      *         an output ZIP file.
894      * @return A stream to read the entry data from or {@code null} if the
895      *         entry does not exist.
896      * @throws ZipAuthenticationException If the entry is encrypted and
897      *         checking the MAC fails.
898      * @throws ZipException If this file is not compatible to the ZIP File
899      *         Format Specification.
900      * @throws IOException If the entry cannot get read from this ZipFile.
901      */
902     @CreatesObligation
903     protected @Nullable InputStream getInputStream(
904             final String name,
905             @CheckForNull Boolean check,
906             final boolean process)
907     throws ZipException, IOException {
908         final SeekableByteChannel channel = channel();
909         Objects.requireNonNull(name);
910         final ZipEntry entry = entries.get(name);
911         if (null == entry) return null;
912         long pos = entry.getOffset();
913         assert UNKNOWN != pos;
914         pos = mapper.map(pos);
915         final MutableBuffer lfh = MutableBuffer
916                 .allocate(LFH_MIN_LEN)
917                 .littleEndian()
918                 .load(channel.position(pos));
919         if (LFH_SIG != lfh.getUInt())
920             throw new ZipException(name + " (expected local file header)");
921         lfh.position(LFH_FILE_NAME_LENGTH_POS);
922         pos += LFH_MIN_LEN
923                 + lfh.getUShort() // file name length
924                 + lfh.getUShort(); // extra field length
925         SeekableByteChannel echannel;
926         try {
927             echannel = new EntryReadOnlyChannel(
928                     pos, entry.getCompressedSize());
929         } catch (RuntimeException e) {
930             throw (ZipException) new ZipException(
931                     name + " (invalid Local File Header, Data Descriptor or Central File Header)")
932                     .initCause(e);
933         }
934         try {
935             if (!process) {
936                 assert UNKNOWN != entry.getCrc();
937                 return new ChannelInputStream(echannel);
938             }
939             if (null == check) check = entry.isEncrypted();
940             int method = entry.getMethod();
941             if (entry.isEncrypted()) {
942                 if (WINZIP_AES != method)
943                     throw new ZipException(name
944                             + " (encrypted compression method "
945                             + method
946                             + " is not supported)");
947                 final WinZipAesReadOnlyChannel
948                         eechannel = new WinZipAesReadOnlyChannel(echannel,
949                                 new WinZipAesEntryParameters(
950                                     parameters(
951                                         WinZipAesParameters.class,
952                                         getCryptoParameters()),
953                                     entry));
954                 echannel = eechannel;
955                 if (check) {
956                     eechannel.authenticate();
957                     // Disable redundant CRC-32 check.
958                     check = false;
959                 }
960                 final WinZipAesExtraField field
961                         = (WinZipAesExtraField) entry.getExtraField(WINZIP_AES_ID);
962                 method = field.getMethod();
963             }
964             if (check) {
965                 // Check CRC32 in the Local File Header or Data Descriptor.
966                 long localCrc;
967                 if (entry.getGeneralPurposeBitFlag(GPBF_DATA_DESCRIPTOR)) {
968                     // The CRC32 is in the Data Descriptor after the compressed
969                     // size.
970                     // Note the Data Descriptor's Signature is optional:
971                     // All newer apps should write it (and so does TrueVFS),
972                     // but older apps might not.
973                     final MutableBuffer dd = MutableBuffer
974                             .allocate(8)
975                             .littleEndian()
976                             .load(channel.position(pos + entry.getCompressedSize()));
977                     localCrc = dd.getUInt();
978                     if (DD_SIG == localCrc) localCrc = dd.getUInt();
979                 } else {
980                     // The CRC32 in the Local File Header.
981                     localCrc = lfh.position(14).getUInt();
982                 }
983                 if (entry.getCrc() != localCrc)
984                     throw new Crc32Exception(name, entry.getCrc(), localCrc);
985             }
986             final int bufSize = getBufferSize(entry);
987             InputStream in;
988             switch (method) {
989                 case STORED:
990                     in = new ChannelInputStream(echannel);
991                     break;
992                 case DEFLATED:
993                     in = new ZipInflaterInputStream(
994                             new DummyByteChannelInputStream(echannel),
995                             bufSize);
996                     break;
997                 case BZIP2:
998                     in = new BZip2CompressorInputStream(
999                             new ChannelInputStream(echannel));
1000                     break;
1001                 default:
1002                     throw new ZipException(name
1003                             + " (compression method "
1004                             + method
1005                             + " is not supported)");
1006             }
1007             if (check) in = new Crc32InputStream(in, bufSize, entry);
1008             return in;
1009         } catch (final Throwable e1) {
1010             try {
1011                 echannel.close();
1012             } catch (final Throwable e2) {
1013                 e1.addSuppressed(e2);
1014             }
1015             throw e1;
1016         }
1017     }
