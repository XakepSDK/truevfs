5   package net.java.truevfs.comp.zip;
6   
7   import edu.umd.cs.findbugs.annotations.CleanupObligation;
8   import edu.umd.cs.findbugs.annotations.CreatesObligation;
9   import edu.umd.cs.findbugs.annotations.DischargesObligation;
10  import java.io.IOException;
11  import java.io.OutputStream;
12  import java.util.zip.ZipException;
13  import javax.annotation.concurrent.NotThreadSafe;
14  
15  /**
16   * A method for writing a ZIP entry.
17   * <p>
18   * Implementations cannot be thread-safe.
19   *
20   * @author  Christian Schlichtherle
21   */
22  @NotThreadSafe
23  @CleanupObligation
24  interface OutputMethod {
25  
26      /**
27       * Checks the given {@code entry} and updates it.
28       * This method may be called multiple times, so it must be idempotent with
29       * respect to its side effects on {@code entry}!
30       * 
31       * @param  entry the ZIP entry to check and update.
32       * @throws ZipException if checking the given entry failed for
33       *         some reason.
34       */
35      void init(ZipEntry entry) throws ZipException;
36  
37      /**
38       * Starts writing the initialized ZIP entry and returns an output stream
39       * for writing its contents.
40       * You must call {@link #finish()} after writing the contents to the
41       * returned output stream.
42       * You must not call {@link OutputStream#close()} on the returned output
43       * stream!
44       * 
45       * @return The decorated output stream.
46       * @throws IOException on any I/O error.
47       */
48      @CreatesObligation
49      OutputStream start() throws IOException;
50  
51      /**
52       * Finishes writing the initialized ZIP entry.
53       *
54       * @throws IOException on any I/O error.
55       */
56      @DischargesObligation
57      void finish() throws IOException;
58  }
