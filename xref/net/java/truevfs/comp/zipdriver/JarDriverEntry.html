5   package net.java.truevfs.comp.zipdriver;
6   
7   import net.java.truevfs.comp.zip.DateTimeConverter;
8   import net.java.truevfs.comp.zip.ZipEntry;
9   
10  /**
11   * JAR driver entries apply the date/time conversion rules as defined by
12   * {@link DateTimeConverter#JAR}.
13   *
14   * @see    #getDateTimeConverter()
15   * @see    JarDriver
16   * @author Christian Schlichtherle
17   */
18  public class JarDriverEntry extends AbstractZipDriverEntry {
19  
20      public JarDriverEntry(String name) {
21          super(name);
22      }
23  
24      protected JarDriverEntry(String name, ZipEntry template) {
25          super(name, template);
26      }
27  
28      /**
29       * {@inheritDoc}
30       * <p>
31       * The implementation in the class {@link JarDriverEntry} returns
32       * {@link DateTimeConverter#JAR}.
33       *
34       * @return {@link DateTimeConverter#JAR}
35       */
36      @Override
37      protected DateTimeConverter getDateTimeConverter() {
38          return DateTimeConverter.JAR;
39      }
40  }
