5   package net.java.truevfs.comp.inst;
6   
7   import java.io.IOException;
8   import java.io.InputStream;
9   import java.nio.channels.SeekableByteChannel;
10  import java.util.Objects;
11  import javax.annotation.CheckForNull;
12  import javax.annotation.concurrent.Immutable;
13  import net.java.truecommons.cio.DecoratingInputSocket;
14  import net.java.truecommons.cio.Entry;
15  import net.java.truecommons.cio.InputSocket;
16  import net.java.truecommons.cio.OutputSocket;
17  
18  /**
19   * @param  <M> the type of the mediator.
20   * @param  <E> the type of the {@linkplain #target() target entry} for I/O
21   *         operations.
22   * @see    InstrumentingOutputSocket
23   * @author Christian Schlichtherle
24   */
25  @Immutable
26  public class InstrumentingInputSocket<
27          M extends Mediator<M>,
28          E extends Entry>
29  extends DecoratingInputSocket<E> {
30  
31      protected final M mediator;
32  
33      public InstrumentingInputSocket(
34              final M mediator,
35              final InputSocket<? extends E> socket) {
36          super(socket);
37          this.mediator = Objects.requireNonNull(mediator);
38      }
39  
40      @Override
41      public InputStream stream(@CheckForNull OutputSocket<? extends Entry> peer)
42      throws IOException {
43          return mediator.instrument(this, socket.stream(peer));
44      }
45  
46      @Override
47      public SeekableByteChannel channel(
48              @CheckForNull OutputSocket<? extends Entry> peer)
49      throws IOException {
50          return mediator.instrument(this, socket.channel(peer));
51      }
52  }
