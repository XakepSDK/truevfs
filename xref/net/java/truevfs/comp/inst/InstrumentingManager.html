5   package net.java.truevfs.comp.inst;
6   
7   import java.util.Objects;
8   import javax.annotation.concurrent.ThreadSafe;
9   import net.java.truevfs.kernel.spec.*;
10  
11  /**
12   * @param  <M> the type of the mediator.
13   * @author Christian Schlichtherle
14   */
15  @ThreadSafe
16  public class InstrumentingManager<M extends Mediator<M>>
17  extends FsDecoratingManager {
18  
19      protected final M mediator;
20  
21      public InstrumentingManager(
22              final M mediator,
23              final FsManager manager) {
24          super(manager);
25          this.mediator = Objects.requireNonNull(mediator);
26      }
27  
28      @Override
29      public FsController controller(
30              FsCompositeDriver driver,
31              FsMountPoint mountPoint) {
32          return mediator.instrument(this,
33                  manager.controller(
34                      mediator.instrument(this, driver),
35                      mountPoint));
36      }
37  }
