7   import java.io.IOException;
8   import javax.annotation.concurrent.ThreadSafe;
9   
10  /**
11   * Indicates that there is an issue when reading or writing a TAR file which
12   * is specific to the TAR file format.
13   * 
14   * @author Christian Schlichtherle
15   */
16  @ThreadSafe
17  public class TarException extends IOException {
18      private static final long serialVersionUID = 8928709487265227047L;
19  
20      public TarException(String message) {
21          super(message);
22      }
23  
24      public TarException(String message, Throwable cause) {
25          super(message, cause);
26      }
27  }
