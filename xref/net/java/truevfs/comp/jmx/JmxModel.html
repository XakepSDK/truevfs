5   package net.java.truevfs.comp.jmx;
6   
7   import net.java.truevfs.comp.inst.InstrumentingModel;
8   import net.java.truevfs.kernel.spec.FsModel;
9   
10  import javax.annotation.concurrent.ThreadSafe;
11  import javax.management.ObjectName;
12  
13  /**
14   * A controller for a {@linkplain FsModel file system model}.
15   *
16   * @param  <M> the type of the JMX mediator.
17   * @author Christian Schlichtherle
18   */
19  @ThreadSafe
20  public class JmxModel<M extends JmxMediator<M>>
21  extends InstrumentingModel<M> implements JmxComponent {
22  
23      private final ObjectName objectName;
24  
25      public JmxModel(M mediator, FsModel model) {
26          super(mediator, model);
27          this.objectName = objectName();
28      }
29  
30      private ObjectName objectName() {
31          return mediator.nameBuilder(FsModel.class)
32                  .put("mountPoint", ObjectName.quote(
33                      model.getMountPoint().toHierarchicalUri().toString()))
34                  .get();
35      }
36  
37      protected Object newView() { return new JmxModelView<>(model); }
38  
39      @Override
40      public void activate() { }
41  
42      @Override
43      public void setMounted(final boolean mounted) {
44          if (model.isMounted() != mounted) {
45              model.setMounted(mounted);
46              if (mounted)
47                  mediator.register(objectName, newView());
48              else
49                  mediator.deregister(objectName);
50          }
51      }
52  }
