5   package net.java.truevfs.comp.jmx;
6   
7   import net.java.truecommons.jmx.ObjectNameBuilder;
8   import net.java.truecommons.jmx.sl.MBeanServerLocator;
9   import net.java.truecommons.logging.LocalizedLogger;
10  import net.java.truevfs.comp.inst.Mediator;
11  import org.slf4j.Logger;
12  
13  import javax.annotation.CheckForNull;
14  import javax.annotation.concurrent.Immutable;
15  import javax.management.*;
16  
17  /**
18   * A mediator for the instrumentation of the TrueVFS Kernel with JMX.
19   *
20   * @param  <This> the type of this mediator.
21   * @author Christian Schlichtherle
22   */
23  @Immutable
24  public abstract class JmxMediator<This extends JmxMediator<This>>
25  extends Mediator<This> {
26  
27      /**
28       * The localized logger to use.
29       * This field is available for unit testing purposes only!
30       */
31      private static final Logger logger = new LocalizedLogger(JmxMediator.class);
32  
33      /**
34       * Returns the package object who's name should be used as the domain of
35       * any object names to register.
36       */
37      public Package getDomain() { return getClass().getPackage(); }
38  
39      /**
40       * Returns a new object name builder.
41       *
42       * @param type the class identifying the type of the MBean who's name shall
43       *        get build.
44       */
45      public ObjectNameBuilder nameBuilder(Class<?> type) {
46          return new ObjectNameBuilder(getDomain())
47                  .put("type", type.getSimpleName());
48      }
49  
50      /**
51       * Returns the MBean server for registering the MBeans.
52       * <p>
53       * The implementation in the class {@link JmxMediator} returns
54       * {@code MBeanServerLocator.SINGLETON.get()}.
55       *
56       * @return the MBean server for registering the MBeans.
57       */
58      public MBeanServer getMBeanServer() {
59          return MBeanServerLocator.SINGLETON.get();
60      }
61  
62      /**
63       * {@linkplain JmxComponent#activate Activates} and returns the given
64       * {@code component}.
65       *
66       * @param  <C> the type of the component to activate.
67       * @param  component the component to activate.
68       * @return The activated component.
69       */
70      public <C extends JmxComponent> C activate(C component) {
71          component.activate();
72          return component;
73      }
74  
75      /**
76       * Maps the given object {@code name} to the given {@code mbean}
77       * in the {@linkplain #getMBeanServer MBean server}.
78       * <p>
79       * The implementation in the class {@link JmxMediator} logs error
80       * messages at the warn and trace level when the registration fails with
81       * an {@link InstanceAlreadyExistsException}.
82       *
83       * @param  name the object name.
84       * @param  mbean the MBean.
85       * @return {@code true} if the MBean has been successfully registered
86       *         with the given {@code name}.
87       *         {@code false} if any MBean was already registered
88       *         with the given {@code name}.
89       * @throws IllegalArgumentException if registering the MBean failed with an
90       *         {@link JMException}.
91       */
92      public boolean register(
93              final @CheckForNull ObjectName name,
94              final Object mbean) {
95          try {
96              getMBeanServer().registerMBean(mbean, name);
97              return true;
98          } catch (final InstanceAlreadyExistsException ex) {
99              logger.warn("instanceAlreadyExists.warn", name);
100             logger.trace("instanceAlreadyExists.trace", ex);
101             return false;
102         } catch (final JMException ex) {
103             throw new IllegalArgumentException(ex);
104         }
105     }
106 
107     /**
108      * Removes any MBean with the given object {@code name}
109      * in the configured MBean server.
110      * <p>
111      * The implementation in the class {@link JmxMediator} logs error
112      * messages at the warn and trace level when the deregistration fails with
113      * an {@link InstanceNotFoundException}.
114      *
115      * @param  name the object name.
116      * @return {@code true} if any MBean has been successfully removed
117      *         with the given {@code name}.
118      *         {@code false} if no MBean was registered
119      *         with the given {@code name}.
120      * @throws IllegalArgumentException if removing an MBean failed with an
121      *         {@link JMException}.
122      */
123     public boolean deregister(final ObjectName name) {
124         try {
125             getMBeanServer().unregisterMBean(name);
126             return true;
127         } catch (final InstanceNotFoundException ex) {
128             logger.warn("instanceNotFound.warn", name);
129             logger.trace("instanceNotFound.trace", ex);
130             return false;
131         } catch (final JMException ex) {
132             throw new IllegalArgumentException(ex);
133         }
134     }
135 }
