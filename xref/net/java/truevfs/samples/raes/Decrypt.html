5   package net.java.truevfs.samples.raes;
6   
7   import java.io.IOException;
8   import net.java.truevfs.access.TPath;
9   
10  /**
11   * Decrypts the contents of the RAES file provided as the first argument
12   * into the file provided as the second argument for the main method.
13   * <p>
14   * Please note that you should not use this utility to decrypt an RAES
15   * encrypted ZIP file (usually a file with a {@code ".tzp"} or
16   * {@code ".zip.rae"} or {@code ".zip.raes"} extension) back to a plain ZIP
17   * file.
18   * This is because RAES encrypted ZIP files use the &quot;UTF-8&quot;
19   * as their character set, whereas plain ZIP files use &quot;IBM437&quot;,
20   * a.k.a. &quot;CP437&quot;.
21   *
22   * @author Christian Schlichtherle
23   */
24  public class Decrypt extends Application {
25      
26      public static void main(String[] args) throws IOException {
27          System.exit(new Decrypt().run(args));
28      }
29  
30      @Override
31      void runChecked(final String[] args) throws IOException {
32          if (args.length != 2) throw new IllegalArgumentException();
33          Raes.decrypt(   new TPath(args[0]).toNonArchivePath(),
34                          new TPath(args[1]).toNonArchivePath(),
35                          true);
36      }
37  }
