5   package net.java.truevfs.driver.tar.bzip2;
6   
7   import java.util.Map;
8   import javax.annotation.concurrent.Immutable;
9   import net.java.truecommons.annotations.ServiceImplementation;
10  import net.java.truecommons.shed.ExtensionSet;
11  import net.java.truevfs.kernel.spec.FsDriver;
12  import net.java.truevfs.kernel.spec.FsScheme;
13  import net.java.truevfs.kernel.spec.spi.FsDriverMapModifier;
14  
15  /**
16   * Maps a file system driver for accessing the BZIP2 compressed TAR file format.
17   * The modified map will contain the following entries:
18   * <p>
19  <table border=1 cellpadding=5 summary="">
20  <thead>
21  <tr>
22  <th>URI Schemes / Archive File Extensions</th>
23  <th>File System Driver Class</th>
24  </tr>
25  </thead>
26  <tbody>
27  <tr>
28  <td>{@code tar.bz2}, {@code tar.bzip2}, {@code tb2}, {@code tbz}, {@code tbz2}</td>
29  <td>{@link TarBZip2Driver}</td>
30  </tr>
31  </tbody>
32  </table>
33   *
34   * @author Christian Schlichtherle
35   */
36  @Immutable
37  @ServiceImplementation
38  public final class TarBZip2DriverMapModifier extends FsDriverMapModifier {
39  
40      @Override
41      public Map<FsScheme, FsDriver> apply(final Map<FsScheme, FsDriver> map) {
42          final FsDriver driver = new TarBZip2Driver();
43          for (final String extension : new ExtensionSet("tar.bz2|tar.bzip2|tb2|tbz|tbz2"))
44              map.put(FsScheme.create(extension), driver);
45          return map;
46      }
47  
48      /** @return -100 */
49      @Override
50      public int getPriority() { return -100; }
51  }
