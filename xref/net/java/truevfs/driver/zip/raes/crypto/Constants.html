5   package net.java.truevfs.driver.zip.raes.crypto;

7   /**
8    * This interfaces contains constants used to read or write files
9    * according to the Random Access Encryption Specification (Constants).
10   * Public classes <em>must not</em> implement this interface - otherwise the
11   * constants become part of the public API.
12   *
13   * @author  Christian Schlichtherle
14   */
15  interface Constants {
16  
17      /**
18       * The signature of any Constants compatible content in little endian format
19       * ({@code "Constants"} as a US-ASCII character sequence).
20       */
21      int SIGNATURE = 'R' | (('A' | (('E' | ('S' << 8)) << 8)) << 8);
22  
23      int HEADER_MIN_LEN =
24              4 + // SIGNATURE
25              1;  // Envelope type
26  
27      /**
28       * The data envelope type used for password based encryption
29       * with the same salt length as the cipher key length.
30       */
31      byte TYPE_0 = 0;
32  
33      /** The length of the header before the salt and the encrypted data. */
34      int TYPE_0_HEADER_LEN_WO_SALT =
35              HEADER_MIN_LEN +
36              1 + // Cipher and MAC key strength.
37              2;  // Iteration count
38                  // The salt which's length is the cipher key length.
39                  // The ciphered data which has the same length as the plain data.
40                  // The KLAC (first half of 256 bit SHA output = 128 bits).
41                  // The  MAC (first half of 256 bit SHA output = 128 bits).
42  
43      /**
44       * The block size of the Advanced Encryption Specification (AES) Algorithm
45       * in bits ({@value #AES_BLOCK_SIZE_BITS}).
46       */
47      int AES_BLOCK_SIZE_BITS = 128;
48  }
