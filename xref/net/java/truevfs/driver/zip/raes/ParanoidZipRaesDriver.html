5   package net.java.truevfs.driver.zip.raes;

7   import java.io.IOException;
8   import javax.annotation.CheckForNull;
9   import javax.annotation.WillNotClose;
10  import javax.annotation.concurrent.Immutable;
11  import net.java.truevfs.comp.zipdriver.JarDriverEntry;
12  import net.java.truevfs.comp.zipdriver.ZipInputService;
13  import net.java.truevfs.comp.zipdriver.ZipOutputService;
14  import net.java.truevfs.kernel.spec.FsModel;
15  import net.java.truevfs.kernel.spec.FsOutputSocketSink;
16  import net.java.truecommons.cio.InputService;
17  
18  /**
19   * A paranoid archive driver for RAES encrypted ZIP files.
20   * This driver <em>always</em> checks the cipher text of input archive files
21   * using the RAES Message Authentication Code (MAC), which makes it slower than
22   * the {@link SafeZipRaesDriver} for archive files larger than 512 KB and
23   * may pause the client application on the first access to the archive file
24   * for a while if the file is large.
25   * Note that the CRC-32 value of the plain text ZIP file is never checked
26   * because this is made redundant by the MAC verification.
27   * <p>
28   * In addition, this driver limits the number of concurrent entry sink
29   * streams to one, so that writing unencrypted temporary files is inhibited.
30   * <p>
31   * Subclasses must be thread-safe and should be immutable!
32   * 
33   * @see    SafeZipRaesDriver
34   * @author Christian Schlichtherle
35   */
36  @Immutable
37  public class ParanoidZipRaesDriver extends ZipRaesDriver {
38  
39      @Override
40      public final long getAuthenticationTrigger() {
41          return Long.MAX_VALUE;
42      }
43  
44      /**
45       * {@inheritDoc}
46       * <p>
47       * The implementation in the class {@link ParanoidZipRaesDriver} returns a
48       * new {@link ZipOutputService}.
49       * This restricts the number of concurrent sink entry streams to one in
50       * order to inhibit writing unencrypted temporary files for buffering the
51       * written entries.
52       */
53      @Override
54      protected final ZipOutputService<JarDriverEntry> newOutput(
55              final FsModel model,
56              final FsOutputSocketSink sink,
57              final @CheckForNull @WillNotClose InputService<JarDriverEntry> input)
58      throws IOException {
59          final ZipInputService<JarDriverEntry> zis = (ZipInputService<JarDriverEntry>) input;
60          return new ZipOutputService<>(model, new RaesSocketSink(model, sink), zis, this);
61      }
62  }
