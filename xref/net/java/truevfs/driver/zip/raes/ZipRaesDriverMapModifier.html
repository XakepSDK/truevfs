8   import javax.annotation.concurrent.Immutable;
9   import net.java.truecommons.annotations.ServiceImplementation;

15  /**
16   * Maps a file system driver for accessing the RAES encrypted ZIP file format,
17   * alias ZIP.RAES or TZP.
18   * The modified map will contain the following entries:
19   * <p>
20  <table border=1 cellpadding=5 summary="">
21  <thead>
22  <tr>
23  <th>URI Schemes / Archive File Extensions</th>
24  <th>File System Driver Class</th>
25  </tr>
26  </thead>
27  <tbody>
28  <tr>
29  <td>{@code tzp}, {@code zip.rae}, {@code zip.raes}</td>
30  <td>{@link SafeZipRaesDriver}</td>
31  </tr>
32  </tbody>
33  </table>
34   *
35   * @author  Christian Schlichtherle
36   */
37  @Immutable
38  @ServiceImplementation
39  public final class ZipRaesDriverMapModifier extends FsDriverMapModifier {
40  
41      @Override
42      public Map<FsScheme, FsDriver> apply(final Map<FsScheme, FsDriver> map) {
43          // Select exactly ONE of the following drivers by
44          // uncommenting it.
45          // Please refer to the class Javadoc for more information.
46  
47          // If you're quite paranoid, then this driver is for you:
48          // It authenticates every input archive file using
49          // the Message Authentication Code (MAC) specified by the
50          // RAES file format, which makes it comparably slow.
51          // The driver also uses unencrypted byte arrays for
52          // temporary storage of archive entries whenever required.
53          // If you were completely paranoid, you would even want to
54          // use encrypted byte arrays or wipe them with nulls after
55          // use.
56          // However, then you would have to write this yourself! ;-)
57          /*final FsDriver driver = new ParanoidZipRaesDriver(new ByteArrayIOPoolProvider(2048));*/
58  
59          // If you're just a bit paranoid, then use this driver:
60          // It authenticates every input archive file using
61          // the Message Authentication Code (MAC) specified by the
62          // RAES file format, which makes it comparably slow.
63          // The driver also uses unencrypted temporary files for
64          // archive entries whenever required.
65          /*final FsDriver driver = new ParanoidZipRaesDriver(IOPoolLocator.SINGLETON);*/
66  
67          // For the rest of us, this driver is our choice:
68          // It authenticates input archive files up to 512 KB using
69          // the Message Authentication Code (MAC) specified by the
70          // RAES file format.
71          // For larger input archive files, it just checks the
72          // CRC-32 value whenever an archive entry input stream is
73          // closed.
74          // CRC-32 has frequent collisions when compared to a MAC.
75          // However, it should not be feasible to make an
76          // undetectable modification.
77          // The driver also uses unencrypted temporary files for
78          // archive entries whenever required.
79          final FsDriver driver = new SafeZipRaesDriver();
80  
81          for (final String extension : new ExtensionSet("tzp|zip.rae|zip.raes"))
82              map.put(FsScheme.create(extension), driver);
83          return map;
84      }
85  
86      /** @return -100 */
87      @Override
88      public int getPriority() { return -100; }
89  }
