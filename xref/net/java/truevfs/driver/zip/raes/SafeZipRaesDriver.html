5   package net.java.truevfs.driver.zip.raes;
6   
7   import java.io.IOException;
8   import javax.annotation.concurrent.Immutable;
9   
10  /**
11   * A safe archive driver for RAES encrypted ZIP files (ZIP.RAES).
12   * For input archive files up to 512 KB, the cipher text gets authenticated
13   * using the RAES provided Message Authentication Code (MAC) <em>before</em>
14   * the archive can be accessed by a client application.
15   * For larger input archive files, the MAC is not used, but instead the
16   * CRC-32 value of the decrypted and deflated archive entries is checked
17   * when the archive entry stream is <em>closed</em> by the client application,
18   * resulting in some {@link IOException}.
19   * <p>
20   * This operation mode is considered to be safe:
21   * Although a formal prove is missing, it should be computationally
22   * infeasible to modify an archive file so that <em>after</em> decryption
23   * of the archive and <em>after</em> inflation (decompression) of an
24   * entry's data its CRC-32 value still matches!
25   * This should hold true even though CRC-32 is not at all a good cryptographic
26   * hash function because of its frequent collisions, its linear sink and
27   * small sink size.
28   * It's the ZIP inflation algorithm which actually comes to our rescue!
29   * <p>
30   * Subclasses must be thread-safe and should be immutable!
31   * 
32   * @see     ParanoidZipRaesDriver
33   * @author  Christian Schlichtherle
34   */
35  @Immutable
36  public class SafeZipRaesDriver extends ZipRaesDriver {
37  
38      private static final long AUTHENTICATION_TRIGGER = 512 * 1024;
39  
40      @Override
41      public final long getAuthenticationTrigger() {
42          return AUTHENTICATION_TRIGGER;
43      }
44  }
