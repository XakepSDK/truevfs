5   package net.java.truevfs.driver.odf;

7   import javax.annotation.concurrent.Immutable;
8   import net.java.truevfs.comp.zipdriver.JarDriverEntry;
9   import net.java.truevfs.comp.zipdriver.ZipInputService;

11  /**
12   * An archive driver for ODF files which checks the CRC32 value for all ZIP
13   * entries in input archives.
14   * The additional CRC32 computation makes this class slower than its super
15   * class.
16   * <p>
17   * If there is a mismatch of the CRC32 values for a ZIP entry in an input
18   * archive, the {@link java.io.InputStream#close} method of the corresponding
19   * stream for the archive entry will throw a
20   * {@link net.java.truevfs.comp.zip.Crc32Exception}.
21   * Other than this, the archive entry will be processed normally.
22   * So if just the CRC32 value for the entry in the archive file has been
23   * modified, you can still read its entire contents.
24   * <p>
25   * Subclasses must be thread-safe and should be immutable!
26   * 
27   * @author Christian Schlichtherle
28   */
29  @Immutable
30  public class CheckedOdfDriver extends OdfDriver {
31  
32      /**
33       * {@inheritDoc}
34       * 
35       * @return {@code true}
36       */
37      @Override
38      public boolean check(JarDriverEntry local, ZipInputService<JarDriverEntry> input) {
39          return true;
40      }
41  }
