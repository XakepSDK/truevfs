23  /**
24   * An archive driver for application archive files according to the Open
25   * Document Format (ODF) Specification, V1.0 and later.
26   * This driver simply ensures that an entry named {@code mimetype}, if present
27   * at all, is always written as the very first entry using the method
28   * {@code STORED} rather than {@code DEFLATED} in the resulting archive file
29   * in order to meet the requirements of the ODF Specification.
30   * Other than this, ODF files are treated like regular JAR files.
31   * In particular, this class does <em>not</em> check an ODF file for the
32   * existence of the {@code META-INF/manifest.xml} entry or any other entry.
33   * <p>
34   * When using this driver to create or modify an ODF file, then in order to
35   * achieve best performance, the {@code mimetype} entry should always get
36   * written first in order to avoid temp file buffering of all other entries.
37   * <p>
38   * Subclasses must be thread-safe and should be immutable!
39   *
40   * @see    <a href="http://docs.oasis-open.org/office/v1.0/OpenDocument-v1.0-os.pdf">Open Document Format for Office Applications (OpenDocument) v1.0; Section 17.4: MIME Type Stream</a>
41   * @see    <a href="http://docs.oasis-open.org/office/v1.1/OS/OpenDocument-v1.1.pdf">Open Document Format for Office Applications (OpenDocument) v1.1; Section 17.4: MIME Type Stream</a>
42   * @see    <a href="http://docs.oasis-open.org/office/v1.2/OpenDocument-v1.2-part3.pdf">Open Document Format for Office Applications (OpenDocument) Version 1.2; Part 3: Packages; Section 3.3: MIME Type Stream</a>
43   * @see    OdfOutputService
44   * @author Christian Schlichtherle
45   */
46  @Immutable
47  public class OdfDriver extends JarDriver {
48  
49      @Override
50      protected OutputService<JarDriverEntry> newOutput(
51              final FsModel model,
52              final FsOutputSocketSink sink,
53              final @CheckForNull @WillNotClose InputService<JarDriverEntry> input)
54      throws IOException {
55          final ZipInputService<JarDriverEntry> zis = (ZipInputService<JarDriverEntry>) input;
56          final ZipOutputService<JarDriverEntry> zos = new ZipOutputService<>(model, sink, zis, this);
57          final IoBufferPool pool = getPool();
58          return null != zis && sink.getOptions().get(GROW)
59                  ? new MultiplexingOutputService<>(pool, zos)
60                  : new OdfOutputService(pool, zos);
61      }
62  }
