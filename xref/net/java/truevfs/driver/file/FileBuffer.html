5   package net.java.truevfs.driver.file;
6   
7   import java.io.IOException;
8   import static java.nio.file.Files.deleteIfExists;
9   import java.nio.file.Path;
10  import javax.annotation.CheckForNull;
11  import javax.annotation.concurrent.NotThreadSafe;
12  
13  /**
14   * A pooled file buffer.
15   *
16   * @see    FileBufferPool
17   * @author Christian Schlichtherle
18   */
19  @NotThreadSafe
20  final class FileBuffer extends FileNode {
21  
22      FileBuffer(final Path path, final FileBufferPool pool) {
23          super(path);
24          assert null != pool;
25          this.pool = pool;
26      }
27  
28      @Override public void release() throws IOException { pool(null); }
29  
30      private void pool(@CheckForNull final FileBufferPool newPool) throws IOException {
31          final FileBufferPool oldPool = this.pool;
32          this.pool = newPool;
33          if (oldPool != newPool) deleteIfExists(getPath());
34      }
35  
36      @Override
37      @SuppressWarnings(value = "FinalizeDeclaration")
38      protected void finalize() throws Throwable {
39          try { pool(null); }
40          finally { super.finalize(); }
41      }
42  }
