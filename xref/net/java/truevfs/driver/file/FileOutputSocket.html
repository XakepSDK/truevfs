50  /**
51   * An output socket for a file entry.
52   *
53   * @see    FileInputSocket
54   * @author Christian Schlichtherle
55   */
56  @NotThreadSafe
57  final class FileOutputSocket extends AbstractOutputSocket<FileNode> {
58  
59      private static final int
60              INITIAL_CAPACITY = initialCapacity(FsAccessOption.values().length);
61      private static final StandardOpenOption[]
62              WRITE_STANDARD_OPEN_OPTION = {
63                  StandardOpenOption.WRITE,
64                  StandardOpenOption.TRUNCATE_EXISTING,
65                  StandardOpenOption.CREATE,
66              };
67  
68      private final BitField<FsAccessOption> options;
69      private final FileNode node;
70      private final @CheckForNull Entry template;
71  
72      FileOutputSocket(
73              final BitField<FsAccessOption> options,
74              final FileNode node,
75              final @CheckForNull Entry template) {
76          assert null != node;
77          this.node = node;
78          if (options.get(EXCLUSIVE) && options.get(APPEND))
79              throw new IllegalArgumentException();
80          this.options = options;
81          this.template = template;
82      }
83  
84      @Override
85      public FileNode target() { return node; }
86  
87      private FileNode begin() throws IOException {
88          final FileNode buffer;
89          final Path entryFile = node.getPath();
90          Boolean exists = null;
91          if (options.get(EXCLUSIVE) && (exists = exists(entryFile)))
92              throw new FileAlreadyExistsException(node.toString());
93          if (options.get(CACHE)) {
94              // This is obviously NOT properly isolated.
95              if (TRUE.equals(exists)
96                      || null == exists && (exists = exists(entryFile))) {
97                  //if (!isWritable(entryFile)) throw new IOException(...)
98                  entryFile   .getFileSystem()
99                              .provider()
100                             .checkAccess(entryFile, AccessMode.WRITE);
101             } else {
102                 createFile(entryFile);
103             }
104             buffer = node.createIoBuffer();
105         } else {
106             buffer = node;
107         }
108         if (options.get(CREATE_PARENTS) && !TRUE.equals(exists)) {
109             final Path parentFile = entryFile.getParent();
110             if (null != parentFile) {
111                 try {
112                     createDirectories(parentFile);
113                 } catch (IOException ex) {
114                     // Workaround for bug in Oracle JDK 1.7.0_51:
115                     // A call to Files.createDirectories("C:\\") will fail on
116                     // Windows although it shouldn't.
117                     // Likewise, a call to Files.createDirectories("/") will
118                     // fail on Mac OS X although it shouldn't.
119                     // On Linux, it supposedly works however.
120                     if (!isDirectory(parentFile)) throw ex;
121                 }
122             }
123         }
124         return buffer;
125     }
126 
127     void append(final FileNode buffer) throws IOException {
128         if (buffer != node && options.get(APPEND) && exists(node.getPath()))
129             IoSockets.copy(node.input(), buffer.output());
130     }
131 
132     Set<OpenOption> optionSet() {
133         final Set<OpenOption> set = new HashSet<>(INITIAL_CAPACITY);
134         Collections.addAll(set, WRITE_STANDARD_OPEN_OPTION);
135         if (options.get(APPEND)) {
136             set.add(StandardOpenOption.APPEND);
137             set.remove(StandardOpenOption.TRUNCATE_EXISTING);
138         }
139         if (options.get(EXCLUSIVE))
140             set.add(StandardOpenOption.CREATE_NEW);
141         return set;
142     }
143 
144     OpenOption[] optionArray() {
145         final Set<OpenOption> set = optionSet();
146         return set.toArray(new OpenOption[set.size()]);
147     }
148 
149     void close(final FileNode buffer, final boolean commit)
150     throws IOException {
151         final Path entryFile = node.getPath();
152         if (buffer != node) {
153             final Path bufferFile = buffer.getPath();
154             updateProperties(bufferFile);
155             if (commit) {
156                 try {
157                     move(bufferFile, entryFile, REPLACE_EXISTING);
158                 } catch (final IOException ex) {
159                     // Slow:
160                     /*Files.copy(bufferFile, entryFile,
161                             StandardCopyOption.REPLACE_EXISTING);*/
162                     // Fast:
163                     IoSockets.copy(buffer.input(), node.output());
164                     updateProperties(entryFile);
165                 }
166                 buffer.release();
167             } else {
168                 // Leave bufferFile for post-mortem analysis.
169             }
170         } else {
171             updateProperties(entryFile);
172         }
173     }
174 
175     private void updateProperties(final Path file) throws IOException {
176         final Entry template = this.template;
177         if (null == template)
178             return;
179         getFileAttributeView(file, BasicFileAttributeView.class)
180                 .setTimes(  toFileTime(template.getTime(WRITE)),
181                             toFileTime(template.getTime(READ)),
182                             toFileTime(template.getTime(CREATE)));
183     }
184 
185     private static @Nullable FileTime toFileTime(long time) {
186         return UNKNOWN == time ? null : FileTime.fromMillis(time);
187     }
188 
189     IOException release(
190             final IOException ex,
191             final FileNode buffer)
192     throws IOException {
193         try {
194             buffer.release();
195         } catch (final IOException ex2) {
196             ex.addSuppressed(ex2);
197         }
198         return ex;
199     }
200 
201     @Override
202     public SeekableByteChannel channel(final InputSocket<? extends Entry> peer)
203     throws IOException {
204         final FileNode buffer = begin();
205 
206         final class Channel extends IOExceptionSeekableChannel {
207             boolean closed;
208 
209             Channel() throws IOException {
210                 super(newByteChannel(buffer.getPath(), optionSet()));
211             }
212 
213             @Override
214             public void close() throws IOException {
215                 if (closed) return;
216                 super.close();
217                 closed = true;
218                 FileOutputSocket.this.close(buffer, null == exception);
219             }
220         } // Channel
221 
222         try {
223             append(buffer);
224             return new Channel();
225         } catch (IOException ex) {
226             throw release(ex, buffer);
227         }
228     }
229 
230     @Override
231     public OutputStream stream(final InputSocket<? extends Entry> peer)
232     throws IOException {
233         final FileNode buffer = begin();
234 
235         final class Stream extends IOExceptionOutputStream {
236             boolean closed;
237 
238             @CreatesObligation
239             Stream() throws IOException {
240                 super(Files.newOutputStream(buffer.getPath(), optionArray()));
241             }
242 
243             @Override
244             public void close() throws IOException {
245                 if (closed) return;
246                 super.close();
247                 closed = true;
248                 FileOutputSocket.this.close(buffer, null == exception);
249             }
250         } // Stream
251 
252         try {
253             append(buffer);
254             return new Stream();
255         } catch (IOException ex) {
256             throw release(ex, buffer);
257         }
258     }
