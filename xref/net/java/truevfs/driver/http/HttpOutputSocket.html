5   package net.java.truevfs.driver.http;
6   
7   import java.io.IOException;
8   import java.io.OutputStream;
9   import javax.annotation.CheckForNull;
10  import javax.annotation.concurrent.NotThreadSafe;
11  import net.java.truecommons.cio.AbstractOutputSocket;
12  import net.java.truecommons.cio.Entry;
13  import net.java.truecommons.cio.InputSocket;
14  import net.java.truecommons.shed.BitField;
15  import net.java.truevfs.kernel.spec.FsAccessOption;
16  
17  /**
18   * An output socket for HTTP(S) entries.
19   * Right now, this is only a dummy.
20   *
21   * @see    HttpInputSocket
22   * @author Christian Schlichtherle
23   */
24  @NotThreadSafe
25  public class HttpOutputSocket extends AbstractOutputSocket<HttpNode> {
26  
27      private final HttpNode entry;
28  
29      HttpOutputSocket(
30              final BitField<FsAccessOption> options,
31              final HttpNode entry,
32              final @CheckForNull Entry template) {
33          assert null != entry;
34          assert null != options;
35          this.entry    = entry;
36      }
37  
38      @Override
39      public HttpNode target() {
40          return entry;
41      }
42  
43      @Override
44      public OutputStream stream(final InputSocket<? extends Entry> peer)
45      throws IOException {
46          return entry.newOutputStream();
47      }
48  }
