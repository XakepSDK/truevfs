5   package net.java.truevfs.kernel.spec;
6   
7   import java.util.Formatter;
8   import net.java.truecommons.shed.UniqueObject;
9   
10  /**
11   * An abstract file system node is a node which can implement multiple types
12   * and list directory members.
13   *
14   * @since  TrueVFS 0.10
15   * @author Christian Schlichtherle
16   */
17  public abstract class FsAbstractNode
18  extends UniqueObject implements FsNode {
19  
20      @Override
21      public boolean isType(Type type) { return getTypes().is(type); }
22  
23      /**
24       * Returns a string representation of this object for logging and debugging
25       * purposes.
26       */
27      @Override
28      public String toString() {
29          final StringBuilder s = new StringBuilder(256);
30          final Formatter f = new Formatter(s).format("%s@%x[name=%s, types=%s",
31                  getClass().getName(), hashCode(), getName(), getTypes());
32          for (final Size type : ALL_SIZES) {
33              final long size = getSize(type);
34              if (UNKNOWN != size) f.format(", size(%s)=%d", type, size);
35          }
36          for (final Access type : ALL_ACCESS) {
37              final long time = getTime(type);
38              if (UNKNOWN != time) f.format(", time(%s)=%tc", type, time);
39          }
40          return f.format(", members=%s]", getMembers()).toString();
41      }
42  }
