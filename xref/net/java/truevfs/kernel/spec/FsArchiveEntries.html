5   package net.java.truevfs.kernel.spec;
6   
7   import java.util.Formatter;
8   import static net.java.truecommons.cio.Entry.*;
9   
10  /**
11   * Static utility methods for {@link FsArchiveEntry} objects.
12   * 
13   * @author Christian Schlichtherle
14   */
15  public final class FsArchiveEntries {
16  
17      /* Can't touch this - hammer time! */
18      private FsArchiveEntries() { }
19  
20      /**
21       * Returns a string representation of this object for debugging and logging
22       * purposes.
23       * 
24       * @param  e the archive entry
25       * @return A string representation of this object for debugging and logging
26       *         purposes.
27       */
28      public static String toString(final FsArchiveEntry e) {
29          final StringBuilder s = new StringBuilder(256);
30          final Formatter f = new Formatter(s)
31                  .format("%s@%x[name=%s, type=%s",
32                      e.getClass().getName(), e.hashCode(), e.getName(), e.getType());
33          for (Size type : ALL_SIZES) {
34              final long size = e.getSize(type);
35              if (UNKNOWN != size)
36                  f.format(", size(%s)=%d", type, size);
37          }
38          for (Access type : ALL_ACCESS) {
39              final long time = e.getTime(type);
40              if (UNKNOWN != time)
41                  f.format(", time(%s)=%tc", type, time);
42          }
43          return s.append("]").toString();
44      }
45  }
