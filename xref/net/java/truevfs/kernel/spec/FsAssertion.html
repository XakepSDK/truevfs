5   package net.java.truevfs.kernel.spec;
6   
7   import java.lang.annotation.*;
8   import static java.lang.annotation.ElementType.*;
9   
10  /**
11   * Asserts the properties of a virtual file system operation.
12   * <p>
13   * As of TrueVFS 0.10, application level transactions are not supported,
14   * that is, multiple file system operations cannot get composed into a single
15   * application level transaction - support for this feature may be added in a
16   * future version.
17   * <p>
18   * However, individual file system operations do come with assertions about
19   * their atomicity, consistency, isolation and durability - see
20   * {@link FsController}.
21   *
22   * @since  TrueVFS 0.10
23   * @author Christian Schlichtherle
24   */
25  @Documented
26  @Inherited
27  @Retention(RetentionPolicy.SOURCE)
28  @Target({ CONSTRUCTOR, METHOD })
29  public @interface FsAssertion {
30  
31      /**
32       * Does the operation <em>always</em> either completely succeed or
33       * completely fail?
34       */
35      Level atomic() default Level.UNDEFINED;
36  
37      /**
38       * If the VFS has been in a consistent state when the operation starts,
39       * then does it <em>always</em> leave the VFS in a consistent state, too?
40       */
41      Level consistent() default Level.UNDEFINED;
42  
43      /**
44       * Is it <em>always</em> impossible for other threads to see the effects
45       * of any intermediate steps of the operation?
46       */
47      Level isolated() default Level.UNDEFINED;
48  
49      /**
50       * If the operation has written any data, then is the change
51       * <em>always</em> persisted to the top-level file system?
52       */
53      Level durable() default Level.UNDEFINED;
54  
55      /** The property level of a virtual file system operation. */
56      enum Level { UNDEFINED, NOT_APPLICABLE, NO, YES }
57  }
