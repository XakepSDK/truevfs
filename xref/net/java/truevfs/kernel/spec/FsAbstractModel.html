5   package net.java.truevfs.kernel.spec;
6   
7   import java.util.Objects;
8   import javax.annotation.CheckForNull;
9   import javax.annotation.concurrent.ThreadSafe;
10  import net.java.truecommons.shed.UniqueObject;
11  
12  /**
13   * An abstract file system model which does <em>not</em> implement the property
14   * {@code touched}.
15   * <p>
16   * Subclasses should be thread-safe.
17   *
18   * @author Christian Schlichtherle
19   */
20  @ThreadSafe
21  public abstract class FsAbstractModel
22  extends UniqueObject implements FsModel {
23  
24      private final FsMountPoint mountPoint;
25      private @CheckForNull final FsModel parent;
26  
27      protected FsAbstractModel(
28              final FsMountPoint mountPoint,
29              final @CheckForNull FsModel parent) {
30          if (!Objects.equals(mountPoint.getParent(),
31                      (null == parent ? null : parent.getMountPoint())))
32              throw new IllegalArgumentException("Parent/Member mismatch!");
33          this.mountPoint = mountPoint;
34          this.parent = parent;
35      }
36  
37      @Override
38      public final FsMountPoint getMountPoint() { return mountPoint; }
39  
40      @Override
41      public final @CheckForNull FsModel getParent() { return parent; }
42  
43      /**
44       * Returns a string representation of this object for debugging and logging
45       * purposes.
46       */
47      @Override
48      public String toString() {
49          return String.format("%s@%x[mountPoint=%s, parent=%s, mounted=%b]",
50                  getClass().getName(),
51                  hashCode(),
52                  mountPoint,
53                  parent,
54                  isMounted());
55      }
56  }
