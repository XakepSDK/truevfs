5   package net.java.truevfs.kernel.spec.spi;
6   
7   import java.util.LinkedHashMap;
8   import java.util.Map;
9   import javax.annotation.concurrent.Immutable;
10  import net.java.truecommons.annotations.ServiceImplementation;
11  import net.java.truecommons.annotations.ServiceSpecification;
12  import net.java.truecommons.services.LocatableFactory;
13  import net.java.truevfs.kernel.spec.FsDriver;
14  import net.java.truevfs.kernel.spec.FsScheme;
15  import net.java.truevfs.kernel.spec.sl.FsDriverMapLocator;
16  
17  /**
18   * A service for creating maps of file system schemes to file system drivers.
19   * Note that you shouldn't subclass this class for customization.
20   * It solely exists in order to support the
21   * {@link FsDriverMapLocator#SINGLETON}, which will use it to create the
22   * initial driver map which gets subsequently modified by the
23   * {@link FsDriverMapModifier} implementations found on the class path.
24   *
25   * @author Christian Schlichtherle
26   */
27  @Immutable
28  @ServiceSpecification
29  @ServiceImplementation
30  public class FsDriverMapFactory
31  extends LocatableFactory<Map<FsScheme, FsDriver>> {
32  
33      /**
34       * Returns a new empty map for subsequent modification.
35       *
36       * @return A new empty map for subsequent modification.
37       */
38      @Override
39      public Map<FsScheme, FsDriver> get() { return new LinkedHashMap<>(32); }
40  
41      /**
42       * {@inheritDoc}
43       * <p>
44       * If the {@linkplain #getClass() runtime class} of this object is
45       * {@link FsDriverMapFactory}, then {@code -100} gets returned.
46       * Otherwise, zero gets returned.
47       */
48      @Override
49      public int getPriority() {
50          return FsDriverMapFactory.class.equals(getClass()) ? -100 : 0;
51      }
52  }
