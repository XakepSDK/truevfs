14  /**
15   * An abstract service for modifying maps of file system schemes to file
16   * system drivers.
17   * Modifier services are subject to service location by the
18   * {@link FsDriverMapLocator#SINGLETON}.
19   * <p>
20   * If multiple modifier services are locatable on the class path at run
21   * time, they are applied in ascending order of their
22   * {@linkplain #getPriority() priority} so that the result of the modifier
23   * service with the greatest number becomes the result of the entire
24   * modifier chain.
25   *
26   * @author Christian Schlichtherle
27   */
28  @ServiceSpecification
29  public abstract class FsDriverMapModifier
30  extends LocatableModifier<Map<FsScheme, FsDriver>> { }
