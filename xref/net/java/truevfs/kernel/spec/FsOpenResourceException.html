5   package net.java.truevfs.kernel.spec;
6   
7   import java.io.IOException;
8   import javax.annotation.concurrent.ThreadSafe;
9   
10  /**
11   * Indicates that a call to {@link FsController#sync} cannot succeed because
12   * some threads have unclosed I/O resources, e.g. streams or channels.
13   * <p>
14   * This exception should be recoverable, meaning it should be possible to
15   * successfully retry the operation as soon as these I/O resources have been
16   * closed and no other exceptional conditions apply.
17   * 
18   * @author Christian Schlichtherle
19   */
20  @ThreadSafe
21  public class FsOpenResourceException extends IOException {
22      private static final long serialVersionUID = 1L;
23  
24      final int local, total;
25  
26      public FsOpenResourceException(int local, int total) {
27          super("Thread-local / total number of open I/O resources (e.g. streams or channels): %d / %d");
28          this.local = local;
29          this.total = total;
30      }
31  
32      public int getLocal() { return local; }
33  
34      public int getTotal() { return total; }
35  
36      @Override
37      public String getMessage() {
38          return String.format(super.getMessage(), local, total);
39      }
40  }
