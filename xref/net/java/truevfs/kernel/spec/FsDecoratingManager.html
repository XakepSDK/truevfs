5   package net.java.truevfs.kernel.spec;
6   
7   import net.java.truecommons.shed.Filter;
8   import net.java.truecommons.shed.Visitor;
9   
10  import javax.annotation.Nullable;
11  import javax.annotation.concurrent.ThreadSafe;
12  import java.util.Objects;
13  
14  /**
15   * An abstract decorator for a file system manager.
16   *
17   * @author Christian Schlichtherle
18   */
19  @ThreadSafe
20  public abstract class FsDecoratingManager extends FsAbstractManager {
21  
22      /** The decorated file system manager. */
23      protected final FsManager manager;
24  
25      protected FsDecoratingManager(final FsManager manager) {
26          this.manager = Objects.requireNonNull(manager);
27      }
28  
29      @Override
30      public FsModel newModel(
31              FsDriver context,
32              FsMountPoint mountPoint,
33              FsModel parent) {
34          assert null == parent
35                      ? null == mountPoint.getParent()
36                      : parent.getMountPoint().equals(mountPoint.getParent());
37          return manager.newModel(context, mountPoint, parent);
38      }
39  
40      @Override
41      public FsController newController(
42              FsArchiveDriver<? extends FsArchiveEntry> context,
43              FsModel model,
44              @Nullable FsController parent) {
45          assert null == parent
46                      ? null == model.getParent()
47                      : parent.getModel().equals(model.getParent());
48          return manager.newController(context, model, parent);
49      }
50  
51      @Override
52      public FsController controller(
53              FsCompositeDriver driver,
54              FsMountPoint mountPoint) {
55          return manager.controller(driver, mountPoint);
56      }
57  
58      @Override
59      public <X extends Exception, V extends Visitor<? super FsController, X>> V accept(final Filter<? super FsController> filter, V visitor) throws X {
60          return manager.accept(filter, visitor);
61      }
62  
63      /**
64       * Returns a string representation of this object for debugging and logging
65       * purposes.
66       */
67      @Override
68      public String toString() {
69          return String.format("%s@%x[manager=%s]",
70                  getClass().getName(),
71                  hashCode(),
72                  manager);
73      }
74  }
