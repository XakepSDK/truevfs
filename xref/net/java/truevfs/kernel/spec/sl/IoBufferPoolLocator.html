5   package net.java.truevfs.kernel.spec.sl;
6   
7   import javax.annotation.concurrent.Immutable;
8   import net.java.truecommons.cio.IoBufferPool;
9   import net.java.truecommons.services.Container;
10  import net.java.truecommons.services.ServiceLocator;
11  import net.java.truevfs.kernel.spec.spi.IoBufferPoolDecorator;
12  import net.java.truevfs.kernel.spec.spi.IoBufferPoolFactory;
13  
14  /**
15   * A container of the singleton I/O buffer pool.
16   * The I/O buffer pool is created by using a {@link ServiceLocator} to search for
17   * advertised implementations of the factory service specification class
18   * {@link IoBufferPoolFactory}
19   * and the decorator service specification class
20   * {@link IoBufferPoolDecorator}.
21   *
22   * @author Christian Schlichtherle
23   */
24  @Immutable
25  public final class IoBufferPoolLocator implements Container<IoBufferPool> {
26  
27      /** The singleton instance of this class. */
28      public static final IoBufferPoolLocator SINGLETON = new IoBufferPoolLocator();
29  
30      private IoBufferPoolLocator() { }
31  
32      @Override
33      public IoBufferPool get() { return Lazy.pool; }
34  
35      /** A static data utility class used for lazy initialization. */
36      private static final class Lazy {
37          static final IoBufferPool pool
38                  = new ServiceLocator(IoBufferPoolLocator.class)
39                  .factory(IoBufferPoolFactory.class, IoBufferPoolDecorator.class)
40                  .get();
41      }
42  }
