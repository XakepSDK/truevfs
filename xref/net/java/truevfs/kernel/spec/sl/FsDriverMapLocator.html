5   package net.java.truevfs.kernel.spec.sl;
6   
7   import java.util.Collections;
8   import java.util.Map;
9   import javax.annotation.concurrent.Immutable;
10  import net.java.truecommons.services.Container;
11  import net.java.truecommons.services.ServiceLocator;
12  import net.java.truevfs.kernel.spec.FsDriver;
13  import net.java.truevfs.kernel.spec.FsScheme;
14  import net.java.truevfs.kernel.spec.spi.FsDriverMapFactory;
15  import net.java.truevfs.kernel.spec.spi.FsDriverMapModifier;
16  
17  /**
18   * A container of the singleton immutable map of all known file system schemes
19   * to file system drivers.
20   * The map is populated by using a {@link ServiceLocator} to search for advertised
21   * implementations of the factory service specification class
22   * {@link FsDriverMapFactory}
23   * and the modifier service specification class
24   * {@link FsDriverMapModifier}.
25   * 
26   * @author Christian Schlichtherle
27   */
28  @Immutable
29  public final class FsDriverMapLocator
30  implements Container<Map<FsScheme, FsDriver>> {
31  
32      /** The singleton instance of this class. */
33      public static final FsDriverMapLocator SINGLETON = new FsDriverMapLocator();
34  
35      private FsDriverMapLocator() { }
36  
37      @Override
38      public Map<FsScheme, FsDriver> get() { return Lazy.drivers; }
39  
40      /** A static data utility class used for lazy initialization. */
41      private static final class Lazy {
42          static final Map<FsScheme, FsDriver> drivers
43                  = Collections.unmodifiableMap(
44                      new ServiceLocator(FsDriverMapLocator.class)
45                      .factory(FsDriverMapFactory.class, FsDriverMapModifier.class)
46                      .get());
47      }
48  }
