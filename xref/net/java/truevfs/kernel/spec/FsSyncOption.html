5   package net.java.truevfs.kernel.spec;
6   
7   import java.io.IOException;
8   import javax.annotation.concurrent.Immutable;
9   import net.java.truecommons.io.ClosedInputException;
10  import net.java.truecommons.io.ClosedOutputException;
11  import net.java.truecommons.shed.BitField;
12  
13  /**
14   * Defines options for (virtual) file system synchronization.
15   *
16   * @see    FsController#sync(BitField)
17   * @see    FsSyncOptions
18   * @author Christian Schlichtherle
19   */
20  @Immutable
21  public enum FsSyncOption {
22  
23      /**
24       * Suppose there are any open I/O streams or channels for any file system
25       * entries.
26       * Then if this option is set, the respective file system controller waits
27       * until all <em>other</em> threads have closed their I/O resources before
28       * proceeding with the update of the file system.
29       * I/O resources allocated by the <em>current</em> thread are always
30       * ignored.
31       * If the current thread gets interrupted while waiting, it will stop
32       * waiting and proceed normally as if this option wasn't set.
33       * <p>
34       * <strong>WARNING:</strong> If an I/O resource has not been closed because
35       * the client application does not always properly close its streams, even
36       * on an {@link IOException} (which is a common bug in many Java
37       * applications), then the respective file system controller will not
38       * return from the update until the current thread gets interrupted!
39       */
40      WAIT_CLOSE_IO,
41  
42      /**
43       * Suppose there are any open I/O streams or channels for any file system
44       * entries.
45       * Then if this option is set, the respective file system controller
46       * proceeds to update the file system anyway and finally throws an
47       * {@link FsSyncWarningException} with a
48       * {@link FsOpenResourceException} as its cause to indicate that any
49       * subsequent operations on these resources will fail with an
50       * {@link ClosedInputException} or {@link ClosedOutputException}
51       * respectively because they have been forced to close.
52       * <p>
53       * If this option is not set however, the file system is <em>not</em>
54       * updated, but instead an {@link FsSyncException} with a
55       * {@link FsOpenResourceException} as its cause is thrown to indicate that
56       * the application must close all I/O resources first.
57       */
58      FORCE_CLOSE_IO,
59  
60      /**
61       * If this option is set, all pending changes are aborted.
62       * This option is only meaningful immediately before the file system itself
63       * gets deleted and should not of used by client applications.
64       */
65      ABORT_CHANGES,
66  
67      /**
68       * Suppose a file system controller has selectively cached entry contents.
69       * Then if this option is set when the file system gets synchronized,
70       * the entry contents get cleared.
71       * <p>
72       * Note that this option may induce dead locks or even busy loops
73       * when accessing nested archive files in different threads.
74       *
75       * @see <a href="http://java.net/jira/browse/TRUEZIP-268">#TRUEZIP-268</a>
76       * @see <a href="http://java.net/jira/browse/TRUEZIP-269">#TRUEZIP-269</a>
77       */
78      CLEAR_CACHE,
79  }
