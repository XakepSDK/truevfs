5   package net.java.truevfs.kernel.spec;
6   
7   import javax.annotation.concurrent.Immutable;
8   import net.java.truecommons.shed.BitField;
9   import static net.java.truevfs.kernel.spec.FsSyncOption.*;
10  
11  /**
12   * Provides common bit fields of synchronization options.
13   *
14   * @see    FsController#sync(BitField)
15   * @see    FsSyncOption
16   * @author Christian Schlichtherle
17   */
18  @Immutable
19  public final class FsSyncOptions {
20  
21      /** A bit field with no synchronization options set. */
22      public static final BitField<FsSyncOption>
23              NONE = BitField.noneOf(FsSyncOption.class);
24  
25      /**
26       * Forcibly closes all I/O resources (i.e. streams, channels etc) for any
27       * entries of the file system, flushes and clears its selective entry cache,
28       * commits all changes to its parent file system (if any) and makes its
29       * controller eligible for garbage collection unless any strong references
30       * are held by the client application.
31       * This is equivalent to
32       * {@code BitField.of(FsSyncOption.FORCE_CLOSE_IO, FsSyncOption.CLEAR_CACHE)}.
33       * <p>
34       * These options should be used if an application wants to
35       * synchronize all mounted archive files and make sure to clean up
36       * <em>all</em> resources, including the selective entry cache.
37       * Care should be taken not to use these options while any other thread
38       * is still doing I/O to the archive files because otherwise the threads
39       * may not be able to succeed and receive an exception.
40       */
41      public static final BitField<FsSyncOption>
42              UMOUNT = BitField.of(FORCE_CLOSE_IO, CLEAR_CACHE);
43  
44      /**
45       * Waits for all other threads to close their I/O resources (i.e. streams,
46       * channels etc) for any entries of the file system, flushes its selective
47       * entry cache without clearing it and commits all changes to its parent
48       * file system (if any).
49       * This is equivalent to {@code BitField.of(FsSyncOption.WAIT_CLOSE_IO)}.
50       * <p>
51       * These options should be used if a multithreaded application wants to
52       * synchronize all mounted archive files without affecting any I/O to
53       * these archive files by any other thread.
54       * <p>
55       * Note that this bit field deliberately doesn't include CLEAR_CACHE!
56       * This is because CLEAR_CACHE may induce dead locks or even busy loops
57       * when accessing nested archive files in different threads.
58       *
59       * @see <a href="http://java.net/jira/browse/TRUEZIP-268">#TRUEZIP-268</a>
60       * @see <a href="http://java.net/jira/browse/TRUEZIP-269">#TRUEZIP-269</a>
61       */
62      public static final BitField<FsSyncOption>
63              SYNC = BitField.of(WAIT_CLOSE_IO);
64  
65      /**
66       * Aborts all pending changes for the federated file system, clears the
67       * selective cache without flushing it and makes the file system controller
68       * eligible for garbage collection unless any strong references are held by
69       * the client application.
70       * This is equivalent to
71       * {@code BitField.of(FsSyncOption.ABORT_CHANGES)}.
72       * <p>
73       * These options are only meaningful immediately before the federated file
74       * system itself gets deleted and should not of used by client
75       * applications.
76       */
77      public static final BitField<FsSyncOption>
78              RESET = BitField.of(ABORT_CHANGES);
79  
80      /**
81       * Converts the given array to a bit field of synchronization options.
82       *
83       * @param  options an array of synchronization options.
84       * @return A bit field of synchronization options.
85       */
86      public static BitField<FsSyncOption> of(FsSyncOption... options) {
87          return 0 == options.length ? NONE : BitField.of(options[0], options);
88      }
89  
90      /* Can't touch this - hammer time! */
91      private FsSyncOptions() { }
92  }
