5   package net.java.truevfs.access;
6   
7   import edu.umd.cs.findbugs.annotations.CleanupObligation;
8   import edu.umd.cs.findbugs.annotations.CreatesObligation;
9   import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
10  
11  import java.io.File;
12  import java.io.FileReader;
13  import java.io.IOException;
14  import java.io.InputStreamReader;
15  import java.nio.charset.Charset;
16  import java.nio.charset.CharsetDecoder;
17  import javax.annotation.concurrent.Immutable;
18  
19  /**
20   * A replacement for the class {@link FileReader} for reading plain old files
21   * or entries in an archive file.
22   * Mind that applications cannot read archive files directly - just their
23   * entries!
24   *
25   * @see    TFileWriter
26   * @author Christian Schlichtherle
27   */
28  @Immutable
29  @CleanupObligation
30  public final class TFileReader extends InputStreamReader {
31  
32      /**
33       * Constructs a new {@code TFile} reader.
34       * This reader uses the default character set for decoding bytes to
35       * characters.
36       * 
37       * @param  file a file to read.
38       * @throws IOException on any I/O error.
39       */
40      @CreatesObligation
41      @SuppressFBWarnings("DM_DEFAULT_ENCODING")
42      public TFileReader(File file) throws IOException {
43  	super(new TFileInputStream(file));
44      }
45  
46      /**
47       * Constructs a new {@code TFile} reader.
48       * 
49       * @param  file a file to read.
50       * @param  charset a character set for decoding bytes to characters.
51       * @throws IOException on any I/O error.
52       */
53      @CreatesObligation
54      public TFileReader(File file, Charset charset) throws IOException {
55  	super(new TFileInputStream(file), charset);
56      }
57  
58      /**
59       * Constructs a new {@code TFile} reader.
60       * 
61       * @param  file a file to read.
62       * @param  decoder a decoder for decoding bytes to characters.
63       * @throws IOException on any I/O error.
64       */
65      @CreatesObligation
66      public TFileReader(File file, CharsetDecoder decoder) throws IOException {
67  	super(new TFileInputStream(file), decoder);
68      }
69  }
