165     /**
166      * This method returns an image object that can be used to represent the
167      * bean in toolboxes, toolbars, etc. Icon images will typically be GIFs, but
168      * may in future include other formats. <p> Beans aren't required to provide
169      * icons and may return null from this method. <p> There are four possible
170      * flavors of icons (16x16 color, 32x32 color, 16x16 mono, 32x32 mono). If a
171      * bean choses to only support a single icon we recommend supporting 16x16
172      * color. <p> We recommend that icons have a "transparent" background so
173      * they can be rendered onto an existing background.
174      *
175      * @param iconKind The kind of icon requested. This should be one of the
176      * constant values ICON_COLOR_16x16, ICON_COLOR_32x32, ICON_MONO_16x16, or
177      * ICON_MONO_32x32.
178      * @return An image object representing the requested icon. May return null
179      * if no suitable icon is available.
180      */
181     @Override
182     public java.awt.Image getIcon(int iconKind) {
183         switch (iconKind) {
184             case ICON_COLOR_16x16:
185                 if (iconNameC16 == null) {
186                     return null;
187                 } else {
188                     if (iconColor16 == null) {
189                         iconColor16 = loadImage(iconNameC16);
190                     }
191                     return iconColor16;
192                 }
193             case ICON_COLOR_32x32:
194                 if (iconNameC32 == null) {
195                     return null;
196                 } else {
197                     if (iconColor32 == null) {
198                         iconColor32 = loadImage(iconNameC32);
199                     }
200                     return iconColor32;
201                 }
202             case ICON_MONO_16x16:
203                 if (iconNameM16 == null) {
204                     return null;
205                 } else {
206                     if (iconMono16 == null) {
207                         iconMono16 = loadImage(iconNameM16);
208                     }
209                     return iconMono16;
210                 }
211             case ICON_MONO_32x32:
212                 if (iconNameM32 == null) {
213                     return null;
214                 } else {
215                     if (iconMono32 == null) {
216                         iconMono32 = loadImage(iconNameM32);
217                     }
218                     return iconMono32;
219                 }
220             default:
221                 return null;
222         }
223     }
