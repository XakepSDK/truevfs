83      /**
84       * Commits all pending changes for this (federated) file system and all its
85       * federated child file systems to their respective parent file system,
86       * closes their associated target (archive) file in order to allow access
87       * by third parties (e.g.&#160;other processes), cleans up any temporary
88       * allocated resources (e.g.&#160;temporary files) and purges any cached
89       * data.
90       * <p>
91       * Calling this method is equivalent to
92       * {@link #sync(BitField) sync(FsSyncOptions.UMOUNT)}.
93       *
94       * @throws FsSyncWarningException if <em>only</em> warning conditions
95       *         apply.
96       *         This implies that the respective parent file system has been
97       *         synchronized with constraints, e.g. if an unclosed archive entry
98       *         stream gets forcibly closed.
99       * @throws FsSyncException if any error conditions apply.
100      *         This implies some loss of data!
101      * @see    #sync(BitField)
102      */
103     @Override
104     public void close() throws FsSyncWarningException, FsSyncException {
105         sync(UMOUNT);
106     }

108     /**
109      * Commits all pending changes for this (federated) file system and all its
110      * federated child file systems to their respective parent file system with
111      * respect to the given options.
112      *
113      * @param  options a bit field of options for the synchronization operation.
114      * @throws IllegalArgumentException if the combination of synchronization
115      *         options is illegal, e.g. if
116      *         {@code FsSyncOption.FORCE_CLOSE_INPUT} is cleared and
117      *         {@code FsSyncOption.FORCE_CLOSE_OUTPUT} is set or if
118      *         {@code FsSyncOption.ABORT_CHANGES} is set.
119      * @throws FsSyncWarningException if <em>only</em> warning conditions
120      *         apply.
121      *         This implies that the respective parent file system has been
122      *         synchronized with constraints, e.g. if
123      *         {@code FsSyncOption.FORCE_CLOSE_INPUT} or
124      *         {@code FsSyncOption.FORCE_CLOSE_OUTPUT} is set and an unclosed
125      *         archive entry stream gets forcibly closed.
126      * @throws FsSyncException if any error conditions apply.
127      *         This implies some loss of data!
128      */
129     @SuppressWarnings("deprecation")
130     public void sync(BitField<FsSyncOption> options)
131     throws FsSyncWarningException, FsSyncException {
132         TVFS.sync(getMountPoint(), options);
133     }

221     SeekableByteChannel newByteChannel(
222             final TPath path,
223             final Set<? extends OpenOption> options,
224             final FileAttribute<?>... attrs)
225     throws IOException {
226         final FsNodeName name = path.getNodeName();
227         final FsController controller = getController();
228         if (options.isEmpty() || options.contains(StandardOpenOption.READ)) {
229             final BitField<FsAccessOption>
230                     o = path.inputOptions(options).set(CACHE);
231             return controller
232                     .input(o, name)
233                     .channel(null);
234         } else {
235             final BitField<FsAccessOption>
236                     o = path.outputOptions(options).set(CACHE);
237             try {
238                 return controller
239                         .output(o, name, null)
240                         .channel(null);
241             } catch (final IOException ex) {
242                 // TODO: Filter FileAlreadyExistsException.
243                 if (o.get(EXCLUSIVE) && null != controller.node(o, name))
244                     throw (IOException) new FileAlreadyExistsException(path.toString())
245                             .initCause(ex);
246                 throw ex;
247             }
248         }
249     }

265     DirectoryStream<Path> newDirectoryStream(
266             final TPath path,
267             final Filter<? super Path> filter)
268     throws IOException {
269         final FsNode entry = stat(path);
270         final Set<String> set;
271         if (null == entry || null == (set = entry.getMembers()))
272             throw new NotDirectoryException(path.toString());
273 
274         @NotThreadSafe
275         class Adapter implements Iterator<Path> {
276             final Iterator<String> it = set.iterator();
277 
278             @Override
279             public boolean hasNext() {
280                 return it.hasNext();
281             }
282 
283             @Override
284             public Path next() {
285                 return path.resolve(it.next());
286             }
287 
288             @Override
289             public void remove() {
290                 throw new UnsupportedOperationException("Not supported yet.");
291             }
292         } // Adapter
293 
294         @NotThreadSafe
295         class FilterIterator extends FilteringIterator<Path> {
296             FilterIterator() { super(new Adapter()); }
297 
298             @Override
299             protected boolean accept(Path element) {
300                 try {
301                     return filter.accept(element);
302                 } catch (IOException ex) {
303                     throw new DirectoryIteratorException(ex);
304                 }
305             }
306         } // FilterIterator
307 
308         return  new Stream(new FilterIterator());
309     }

311     @NotThreadSafe
312     private static final class Stream implements DirectoryStream<Path> {
313         final Iterator<Path> it;
314         boolean consumed;
315 
316         Stream(final Iterator<Path> it) {
317             this.it = it;
318         }
319 
320         @Override
321         public Iterator<Path> iterator() {
322             if (consumed)
323                 throw new IllegalStateException();
324             consumed = true;
325             return it;
326         }
327 
328         @Override
329         public void close() {
330             consumed = true;
331         }

334     void createDirectory(final TPath path, final FileAttribute<?>... attrs)
335     throws IOException {
336         if (0 < attrs.length)
337             throw new UnsupportedOperationException();
338         final FsController controller = getController();
339         final FsNodeName name = path.getNodeName();
340         final BitField<FsAccessOption> options = path.getAccessPreferences();
341         try {
342             controller.make(
343                     options, name,
344                     DIRECTORY,
345                     null);
346         } catch (IOException ex) {
347             if (null != controller.node(options, name))
348                 throw (IOException) new FileAlreadyExistsException(path.toString())
349                         .initCause(ex);
350             throw ex;
351         }
352     }

373     void checkAccess(final TPath path, final AccessMode... modes)
374     throws IOException {
375         final FsNodeName name = path.getNodeName();
376         final BitField<FsAccessOption> options = path.getAccessPreferences();
377         final BitField<Access> types = types(modes);
378         getController().checkAccess(options, name, types);
379     }
380 
381     private static BitField<Access> types(final AccessMode... modes) {
382         final EnumSet<Access> access = EnumSet.noneOf(Access.class);
383         for (final AccessMode mode : modes) {
384             switch (mode) {
385             case READ:
386                 access.add(READ);
387                 break;
388             case WRITE:
389                 access.add(WRITE);
390                 break;
391             case EXECUTE:
392                 access.add(EXECUTE);
393                 break;
394             }
395         }
396         return BitField.copyOf(access);
397     }

436         @Override
437         public void setTimes(   final FileTime lastModifiedTime,
438                                 final FileTime lastAccessTime,
439                                 final FileTime createTime)
440         throws IOException {
441             final FsController controller = getController();
442             final Map<Access, Long> times = new EnumMap<>(Access.class);
443             if (null != lastModifiedTime)
444                 times.put(WRITE, lastModifiedTime.toMillis());
445             if (null != lastAccessTime)
446                 times.put(READ, lastAccessTime.toMillis());
447             if (null != createTime)
448                 times.put(CREATE, createTime.toMillis());
449             controller.setTime(
450                     path.getAccessPreferences(), path.getNodeName(),
451                     times);
452         }
453     } // FsNodeAttributeView
